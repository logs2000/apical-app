   468→  plan                 String   @default("free") // "free" | "pro" | "enterprise"
   469→  status               String   @default("active") // "active" | "trialing" | "past_due" | "canceled"
   470→  // Stripe customer + subscription ids (null for free / BYOK-only users).

   472→  stripeSubscriptionId String?
   473→  // Price id the user subscribed to (so webhooks can map events → plan).
   474→  stripePriceId        String?
   475→  // Current period end (ISO). Token allowance resets here.
   476→  currentPeriodEnd     DateTime?
   477→  // Monthly token allowance (0 = unlimited for enterprise, or the free quota).
