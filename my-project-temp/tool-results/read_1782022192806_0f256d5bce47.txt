    43→// ─── OS detection (inlined so the page is self-contained) ──────────────────
    44→
    45→type DetectedOS = "mac" | "windows" | "linux" | "other";
    46→
    47→function detectOS(): DetectedOS {
    48→  if (typeof navigator === "undefined") return "other";
    49→  const ua = navigator.userAgent.toLowerCase();
    50→  if (ua.includes("mac os x") || ua.includes("darwin")) return "mac";
    51→  if (ua.includes("windows")) return "windows";
    52→  if (ua.includes("linux")) return "linux";
    53→  return "other";
    54→}
    55→function osLabel(os: DetectedOS): string {
    56→  return os === "mac" ? "macOS" : os === "windows" ? "Windows" : os === "linux" ? "Linux" : "Pick platform";
    57→}
    58→function downloadButtonLabel(os: DetectedOS): string {
    59→  return os === "other" ? "Download" : `Download for ${osLabel(os)}`;
    60→}
    61→function downloadUrl(os: DetectedOS): string {
    62→  const base = "/downloads";
    63→  if (os === "mac") return `${base}/apical-mac.tar.gz`;
    64→  if (os === "windows") return `${base}/apical-windows.exe`;
    65→  if (os === "linux") return `${base}/apical-linux.AppImage`;
    66→  return `${base}/`;
    67→}
    68→function installCommandFor(os: DetectedOS): string {
    69→  if (os === "mac") return "brew install --cask apical";
    70→  if (os === "windows") return "winget install apical.apical";
    71→  if (os === "linux") return "curl -fsSL https://apical.dev/install.sh | sh";
    72→  return "curl -fsSL https://apical.dev/install.sh | sh";
    73→}

   586→
   587→function computePrice(
   588→  plan: Plan,
   589→  interval: "monthly" | "yearly",
   590→): { display: string; suffix?: string; sub?: string } {
   591→  if (plan.id === "enterprise") return { display: "Custom", sub: "Volume pricing — talk to sales." };
   592→  if (plan.id === "free") return { display: "$0", suffix: "/mo", sub: "Free forever. No credit card." };
   593→  const perSeat = plan.id === "team" ? " /seat" : "";
   594→  if (interval === "monthly") {
   595→    return {
   596→      display: `$${plan.priceMonthly}`,
   597→      suffix: `/mo${perSeat}`,
   598→      sub: plan.id === "team" ? "5 seats included." : "Billed monthly.",
   599→    };
   600→  }
   601→  const perMonth = plan.priceYearly / 12;
   602→  const monthsFree = 12 - Math.round(plan.priceYearly / plan.priceMonthly);
   603→  return {
   604→    display: `$${perMonth % 1 === 0 ? perMonth : perMonth.toFixed(2)}`,
   605→    suffix: `/mo${perSeat}`,
   606→    sub: `$${plan.priceYearly}/yr${perSeat} — ${monthsFree} months free`,
   607→  };
   608→}

   714→// ─── Download button + dialog ───────────────────────────────────────────────
   715→
   716→function DownloadButton({
   717→  os,
   718→  variant = "default",
   719→  size = "default",
   720→  className,
   721→}: {
   722→  os: DetectedOS;
   723→  variant?: "default" | "outline" | "ghost";
   724→  size?: "default" | "sm" | "lg";
   725→  className?: string;
   726→}) {
   727→  const [open, setOpen] = React.useState(false);
   728→  const { toast } = useToast();
   729→
   730→  const handleClick = async () => {
   731→    if (os === "other") {
   732→      setOpen(true);
   733→      return;
   734→    }
   735→    // Try the real download URL. If 404, open the dialog with CLI fallback.
   736→    try {
   737→      const res = await fetch(downloadUrl(os), { method: "HEAD" });
   738→      if (res.ok) {
   739→        window.location.href = downloadUrl(os);
   740→        return;
   741→      }
   742→      setOpen(true);
   743→    } catch {
   744→      setOpen(true);
   745→    }
   746→  };
   747→
   748→  return (
   749→    <>
   750→      <Button variant={variant} size={size} onClick={handleClick} className={className}>
   751→        <Download className="mr-1.5 h-4 w-4" />
   752→        {downloadButtonLabel(os)}
   753→      </Button>
   754→      <DownloadDialog open={open} onOpenChange={setOpen} os={os} onCopied={() => toast({ title: "Copied" })} />
   755→    </>
   756→  );
   757→}
   758→
   759→function DownloadDialog({
   760→  open,
   761→  onOpenChange,
   762→  os,
   763→  onCopied,
   764→}: {
   765→  open: boolean;
   766→  onOpenChange: (v: boolean) => void;
   767→  os: DetectedOS;
   768→  onCopied: () => void;
   769→}) {
   770→  const cmd = installCommandFor(os);
   771→  const [copied, setCopied] = React.useState(false);
   772→
   773→  const copy = () => {
   774→    navigator.clipboard?.writeText(cmd).then(() => {
   775→      setCopied(true);
   776→      onCopied();
   777→      setTimeout(() => setCopied(false), 2000);
   778→    });
   779→  };
   780→
   781→  return (
   782→    <Dialog open={open} onOpenChange={onOpenChange}>
   783→      <DialogContent className="sm:max-w-md">
   784→        <DialogHeader>
   785→          <DialogTitle>Install Apical for {osLabel(os)}</DialogTitle>
   786→          <DialogDescription>
   787→            {os === "other"
   788→              ? "Pick your platform below."
   789→              : "The desktop app is the fastest way to get started. If the download above didn't start, use the command below."}
   790→          </DialogDescription>
   791→        </DialogHeader>
   792→
   793→        {os !== "other" && (
   794→          <div className="space-y-3">
   795→            <a
   796→              href={downloadUrl(os)}
   797→              className="flex items-center justify-center gap-2 rounded-lg bg-primary px-4 py-2.5 text-sm font-medium text-primary-foreground transition-opacity hover:opacity-90"
   798→            >
   799→              <Download className="h-4 w-4" /> Download {osLabel(os)} app
   800→            </a>
   801→
   802→            <div className="relative">
   803→              <div className="flex items-center gap-2 rounded-lg border border-border bg-muted/50 p-2.5 pl-3">
   804→                <Terminal className="h-3.5 w-3.5 shrink-0 text-muted-foreground" />
   805→                <code className="flex-1 truncate font-mono text-xs">{cmd}</code>
   806→                <Button variant="ghost" size="icon" className="h-7 w-7 shrink-0" onClick={copy}>
   807→                  {copied ? <Check className="h-3.5 w-3.5 text-primary" /> : <Copy className="h-3.5 w-3.5" />}
   808→                </Button>
   809→              </div>
   810→              <p className="mt-2 text-[11px] text-muted-foreground">
   811→                Or install from the command line.
   812→              </p>
   813→            </div>
   814→          </div>
   815→        )}
   816→
   817→        {os === "other" && (
   818→          <div className="space-y-2">
   819→            {(["mac", "windows", "linux"] as DetectedOS[]).map((o) => (
   820→              <a
   821→                key={o}
   822→                href={downloadUrl(o)}
   823→                className="flex items-center justify-between rounded-lg border border-border p-3 text-sm hover:bg-accent/50"
   824→              >
   825→                <span>{osLabel(o)}</span>
   826→                <Download className="h-4 w-4 text-muted-foreground" />
   827→              </a>
   828→            ))}
   829→          </div>
   830→        )}
   831→
   832→        <DialogFooter className="flex-col gap-2 sm:flex-col sm:space-x-0">
   833→          <Button variant="outline" className="w-full" asChild>
   834→            <a href="#developers">Build from source</a>
   835→          </Button>
   836→        </DialogFooter>
   837→      </DialogContent>
   838→    </Dialog>
   839→  );
   840→}
