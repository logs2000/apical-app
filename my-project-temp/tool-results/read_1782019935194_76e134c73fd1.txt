     1→// Apical LLM gateway — the single routing layer between Apical features
     2→// (chat, agent loops, workflows, research) and every supported model provider.
     3→//
     4→// Three tiers of models (mirroring MODEL_REGISTRY):
     5→//   1. Hosted (apical:*) — Apical-managed. Routes to OpenAI/Anthropic/Google
     6→//      if their *_API_KEY env is set, else falls back to z-ai-web-dev-sdk
     7→//      (the in-house sandbox LLM) which is always available in dev.
     8→//   2. BYOK (user's own key) — user pays the provider directly. costCents = 0
     9→//      (we meter for the dashboard, but don't bill).
    10→//   3. Local (Ollama/llama.cpp) — runs on the user's machine. costCents = 0.
    11→//
    12→// Every call goes through `chat()` (or `chatStream()`) which:
    13→//   - resolves modelId → adapter config (api key, base url, model name)
    14→//   - checks the user's allowance (429-able upstream)
    15→//   - calls the provider
    16→//   - records usage via `recordUsage()` (Subscription + TokenUsageRecord)
    17→//
    18→// All HTTP is plain `fetch`. No axios, no node-fetch. AES-256-GCM via the
    19→// vault module for BYOK keys at rest.
    20→

   118→/** Get-or-create the user's Subscription. A fresh dev user has none yet. */
   119→async function getOrCreateSubscription(userId: string) {
   120→  let sub = await db.subscription.findUnique({ where: { userId } })
   121→  if (!sub) {
   122→    const plan = getPlan('free')
   123→    const periodEnd = new Date()
   124→    periodEnd.setDate(periodEnd.getDate() + 30)
   125→    sub = await db.subscription.create({
   126→      data: {
   127→        userId,
   128→        plan: 'free',
   129→        status: 'active',
   130→        tokenAllowanceMonthly: plan.tokenAllowanceMonthly,
   131→        currentPeriodEnd: periodEnd,
   132→      },
   133→    })
   134→  }
   135→  return sub
   136→}

   140→/**
   141→ * Resolve a modelId to an adapter + credentials.
   142→ *
   143→ * Order:
   144→ *   1. CustomModel row (matched by `id`) — use it. If type=online, load its
   145→ *      ByokKey and decrypt. If type=offline, use its baseUrl.
   146→ *   2. MODEL_REGISTRY entry — byok models load the user's ByokKey for that
   147→ *      provider; hosted `apical:*` models try env keys, falling back to ZAI;
   148→ *      local models use OLLAMA_BASE_URL / LLAMACPP_BASE_URL env.
   149→ *
   150→ * Returns null if the model is unknown or the user lacks a required BYOK key.
   151→ */
   152→export async function resolveModel(
   153→  userId: string,
   154→  modelId: string,
   155→): Promise<ResolvedModel | null> {
   156→  // 1. CustomModel lookup.
   157→  const custom = await db.customModel.findFirst({
   158→    where: { id: modelId, userId, enabled: true },
   159→  })
   160→  if (custom) {
   161→    let apiKey: string | undefined
   162→    let baseUrl: string | undefined
   163→    let adapter: ResolvedModel['adapter'] = 'zai'
   164→    let tier: ModelDefinition['tier'] = 'byok'
   165→
   166→    if (custom.type === 'online') {
   167→      tier = 'byok'
   168→      if (custom.byokKeyId) {
   169→        const byok = await db.byokKey.findUnique({
   170→          where: { id: custom.byokKeyId },
   171→        })
   172→        if (byok && byok.status === 'active') {
   173→          try {
   174→            apiKey = decrypt(byok.encryptedKey)
   175→          } catch {
   176→            apiKey = undefined
   177→          }
   178→          baseUrl = byok.baseUrl ?? undefined
   179→        }
   180→      }
   181→      adapter = pickAdapter(custom.provider as ProviderId)
   182→    } else if (custom.type === 'offline') {
   183→      tier = 'local'
   184→      baseUrl = custom.baseUrl ?? undefined
   185→      adapter = custom.provider === 'llamacpp' ? 'llamacpp' : 'ollama'
   186→    } else {
   187→      // type === 'hosted'
   188→      tier = 'hosted'
   189→      const env = pickHostedEnv(custom.provider as ProviderId)
   190→      if (env) {
   191→        apiKey = env.apiKey
   192→        baseUrl = env.baseUrl
   193→        adapter = env.adapter
   194→      } else {
   195→        adapter = 'zai'
   196→      }
   197→    }
   198→
   199→    const model: ModelDefinition = {
   200→      id: custom.id,
   201→      name: custom.name,
   202→      provider: custom.provider as ProviderId,
   203→      tier,
   204→      apiModelId: custom.modelId,
   205→      contextWindow: custom.contextWindow,
   206→      inputCostCentsPer1M: custom.inputCostCentsPer1M,
   207→      outputCostCentsPer1M: custom.outputCostCentsPer1M,
   208→      supportsStreaming: true,
   209→      supportsTools: false,
   210→      supportsVision: false,
   211→      description: 'User-defined model',
   212→    }
   213→    return { model, adapter, apiKey, baseUrl, isCustom: true }
   214→  }
   215→
   216→  // 2. MODEL_REGISTRY lookup.
   217→  const model = getModel(modelId)
   218→  if (!model) return null
   219→
   220→  if (model.tier === 'hosted') {
   221→    if (model.provider === 'apical') {
   222→      // For apical:* models, prefer a real provider key if set; else ZAI.
   223→      const env = pickHostedEnv('openai') ||
   224→        pickHostedEnv('anthropic') ||
   225→        pickHostedEnv('google')
   226→      if (env) {
   227→        return {
   228→          model: { ...model, provider: env.provider },
   229→          adapter: env.adapter,
   230→          apiKey: env.apiKey,
   231→          baseUrl: env.baseUrl,
   232→        }
   233→      }
   234→      return { model, adapter: 'zai' }
   235→    }
   236→    // Other hosted providers (none currently in registry).
   237→    const env = pickHostedEnv(model.provider)
   238→    if (env) {
   239→      return { model, adapter: env.adapter, apiKey: env.apiKey, baseUrl: env.baseUrl }
   240→    }
   241→    return { model, adapter: 'zai' }
   242→  }
   243→
   244→  if (model.tier === 'byok') {
   245→    const byok = await db.byokKey.findFirst({
   246→      where: { userId, provider: model.provider, status: 'active' },
   247→      orderBy: { updatedAt: 'desc' },
   248→    })
   249→    if (!byok) return null
   250→    let apiKey: string
   251→    try {
   252→      apiKey = decrypt(byok.encryptedKey)
   253→    } catch {
   254→      return null
   255→    }
   256→    return {
   257→      model,
   258→      adapter: pickAdapter(model.provider),
   259→      apiKey,
   260→      baseUrl: byok.baseUrl ?? undefined,
   261→    }
   262→  }
   263→
   264→  // Local.
   265→  if (model.provider === 'ollama') {
   266→    return {
   267→      model,
   268→      adapter: 'ollama',
   269→      baseUrl: process.env.OLLAMA_BASE_URL || 'http://localhost:11434',
   270→    }
   271→  }
   272→  if (model.provider === 'llamacpp') {
   273→    return {
   274→      model,
   275→      adapter: 'llamacpp',
   276→      baseUrl: process.env.LLAMACPP_BASE_URL || 'http://localhost:8080',
   277→    }
   278→  }
   279→  // vLLM exposes an OpenAI-compatible endpoint.
   280→  if (model.provider === 'vllm') {
   281→    return {
   282→      model,
   283→      adapter: 'openai',
   284→      baseUrl: process.env.VLLM_BASE_URL || 'http://localhost:8000/v1',
   285→    }
   286→  }
   287→
   288→  return { model, adapter: 'zai' }
   289→}

   291→function pickAdapter(provider: ProviderId): ResolvedModel['adapter'] {
   292→  switch (provider) {
   293→    case 'openai':
   294→    case 'azure_openai':
   295→    case 'openrouter':
   296→    case 'mistral':
   297→    case 'groq':
   298→    case 'together':
   299→    case 'deepseek':
   300→    case 'vllm':
   301→      return 'openai'
   302→    case 'anthropic':
   303→      return 'anthropic'
   304→    case 'google':
   305→      return 'google'
   306→    case 'ollama':
   307→      return 'ollama'
   308→    case 'llamacpp':
   309→      return 'llamacpp'
   310→    default:
   311→      return 'zai'
   312→  }
   313→}

   315→function pickHostedEnv(provider: ProviderId): {
   316→  provider: ProviderId
   317→  adapter: ResolvedModel['adapter']
   318→  apiKey: string
   319→  baseUrl?: string
   320→} | null {
   321→  switch (provider) {
   322→    case 'openai': {
   323→      const k = process.env.OPENAI_API_KEY
   324→      if (!k) return null
   325→      return {
   326→        provider: 'openai',
   327→        adapter: 'openai',
   328→        apiKey: k,
   329→        baseUrl: process.env.OPENAI_BASE_URL || 'https://api.openai.com/v1',
   330→      }
   331→    }
   332→    case 'anthropic': {
   333→      const k = process.env.ANTHROPIC_API_KEY
   334→      if (!k) return null
   335→      return {
   336→        provider: 'anthropic',
   337→        adapter: 'anthropic',
   338→        apiKey: k,
   339→        baseUrl: process.env.ANTHROPIC_BASE_URL || 'https://api.anthropic.com',
   340→      }
   341→    }
   342→    case 'google': {
   343→      const k = process.env.GOOGLE_API_KEY
   344→      if (!k) return null
   345→      return {
   346→        provider: 'google',
   347→        adapter: 'google',
   348→        apiKey: k,
   349→      }
   350→    }
   351→    default:
   352→      return null
   353→  }
   354→}

   356→// ---------------- Cost computation ----------------
   357→
   358→function computeCost(model: ModelDefinition, usage: ChatUsage): number {
   359→  if (model.tier !== 'hosted') return 0
   360→  const input = (usage.promptTokens * model.inputCostCentsPer1M) / 1_000_000
   361→  const output = (usage.completionTokens * model.outputCostCentsPer1M) / 1_000_000
   362→  return Math.ceil(input + output)
   363→}

   365→// ---------------- Token estimation (for ZAI which doesn't return counts) ----------------
   366→
   367→function estimateUsage(messages: ChatMessage[], completion: string): ChatUsage {
   368→  const promptTokens = Math.ceil(JSON.stringify(messages).length / 4)
   369→  const completionTokens = Math.ceil(completion.length / 4)
   370→  return {
   371→    promptTokens,
   372→    completionTokens,
   373→    totalTokens: promptTokens + completionTokens,
   374→  }
   375→}

   379→/** OpenAI Chat Completions — also used for OpenAI-compatible endpoints. */
   380→async function callOpenAI(
   381→  apiKey: string,
   382→  baseUrl: string,
   383→  model: string,
   384→  messages: ChatMessage[],
   385→  opts: ChatOpts,
   386→): Promise<AdapterResult> {
   387→  const url = `${baseUrl.replace(/\/$/, '')}/chat/completions`
   388→  const body: Record<string, unknown> = {
   389→    model,
   390→    messages,
   391→    stream: false,
   392→  }
   393→  if (typeof opts.maxTokens === 'number') body.max_tokens = opts.maxTokens
   394→  if (typeof opts.temperature === 'number') body.temperature = opts.temperature
   395→
   396→  const res = await fetch(url, {
   397→    method: 'POST',
   398→    headers: {
   399→      'Content-Type': 'application/json',
   400→      Authorization: `Bearer ${apiKey}`,
   401→    },
   402→    body: JSON.stringify(body),
   403→  })
   404→  if (!res.ok) {
   405→    const text = await res.text().catch(() => '')
   406→    throw new Error(`OpenAI API ${res.status}: ${text.slice(0, 500)}`)
   407→  }
   408→  const json = (await res.json()) as {
   409→    choices?: Array<{ message?: { content?: string } }>
   410→    usage?: { prompt_tokens?: number; completion_tokens?: number; total_tokens?: number }
   411→  }
   412→  const content = json.choices?.[0]?.message?.content ?? ''
   413→  const usage: ChatUsage = {
   414→    promptTokens: json.usage?.prompt_tokens ?? 0,
   415→    completionTokens: json.usage?.completion_tokens ?? 0,
   416→    totalTokens: json.usage?.total_tokens ?? 0,
   417→  }
   418→  return { content, usage }
   419→}

   421→/** Anthropic Messages API. */
   422→async function callAnthropic(
   423→  apiKey: string,
   424→  baseUrl: string,
   425→  model: string,
   426→  messages: ChatMessage[],
   427→  opts: ChatOpts,
   428→): Promise<AdapterResult> {
   429→  const url = `${baseUrl.replace(/\/$/, '')}/v1/messages`
   430→  // Anthropic separates `system` from the messages array.
   431→  const systemMsgs = messages.filter((m) => m.role === 'system')
   432→  const userMsgs = messages.filter((m) => m.role !== 'system')
   433→  const body: Record<string, unknown> = {
   434→    model,
   435→    messages: userMsgs,
   436→    max_tokens: opts.maxTokens ?? 1024,
   437→  }
   438→  if (systemMsgs.length) {
   439→    body.system = systemMsgs.map((m) => m.content).join('\n\n')
   440→  }
   441→  if (typeof opts.temperature === 'number') body.temperature = opts.temperature
   442→
   443→  const res = await fetch(url, {
   444→    method: 'POST',
   445→    headers: {
   446→      'Content-Type': 'application/json',
   447→      'x-api-key': apiKey,
   448→      'anthropic-version': '2023-06-01',
   449→    },
   450→    body: JSON.stringify(body),
   451→  })
   452→  if (!res.ok) {
   453→    const text = await res.text().catch(() => '')
   454→    throw new Error(`Anthropic API ${res.status}: ${text.slice(0, 500)}`)
   455→  }
   456→  const json = (await res.json()) as {
   457→    content?: Array<{ type?: string; text?: string }>
   458→    usage?: { input_tokens?: number; output_tokens?: number }
   459→  }
   460→  const content = (json.content ?? [])
   461→    .filter((b) => b.type === 'text')
   462→    .map((b) => b.text ?? '')
   463→    .join('')
   464→  const promptTokens = json.usage?.input_tokens ?? 0
   465→  const completionTokens = json.usage?.output_tokens ?? 0
   466→  return {
   467→    content,
   468→    usage: {
   469→      promptTokens,
   470→      completionTokens,
   471→      totalTokens: promptTokens + completionTokens,
   472→    },
   473→  }
   474→}

   476→/** Google Generative Language API (Gemini). */
   477→async function callGoogle(
   478→  apiKey: string,
   479→  _baseUrl: string | undefined,
   480→  model: string,
   481→  messages: ChatMessage[],
   482→  opts: ChatOpts,
   483→): Promise<AdapterResult> {
   484→  const url = `https://generativelanguage.googleapis.com/v1beta/models/${model}:generateContent?key=${apiKey}`
   485→  // Google's "contents" array doesn't have system role; systemInstruction is separate.
   486→  const systemText = messages
   487→    .filter((m) => m.role === 'system')
   488→    .map((m) => m.content)
   489→    .join('\n\n')
   490→  const contents = messages
   491→    .filter((m) => m.role !== 'system')
   492→    .map((m) => ({
   493→      role: m.role === 'assistant' ? 'model' : 'user',
   494→      parts: [{ text: m.content }],
   495→    }))
   496→  const body: Record<string, unknown> = { contents }
   497→  if (systemText) body.systemInstruction = { parts: [{ text: systemText }] }
   498→  const genConfig: Record<string, unknown> = {}
   499→  if (typeof opts.maxTokens === 'number') genConfig.maxOutputTokens = opts.maxTokens
   500→  if (typeof opts.temperature === 'number') genConfig.temperature = opts.temperature
   501→  if (Object.keys(genConfig).length) body.generationConfig = genConfig
   502→
   503→  const res = await fetch(url, {
   504→    method: 'POST',
   505→    headers: { 'Content-Type': 'application/json' },
   506→    body: JSON.stringify(body),
   507→  })
   508→  if (!res.ok) {
   509→    const text = await res.text().catch(() => '')
   510→    throw new Error(`Google API ${res.status}: ${text.slice(0, 500)}`)
   511→  }
   512→  const json = (await res.json()) as {
   513→    candidates?: Array<{ content?: { parts?: Array<{ text?: string }> } }>
   514→    usageMetadata?: {
   515→      promptTokenCount?: number
   516→      candidatesTokenCount?: number
   517→      totalTokenCount?: number
   518→    }
   519→  }
   520→  const content = (json.candidates?.[0]?.content?.parts ?? [])
   521→    .map((p) => p.text ?? '')
   522→    .join('')
   523→  const promptTokens = json.usageMetadata?.promptTokenCount ?? 0
   524→  const completionTokens = json.usageMetadata?.candidatesTokenCount ?? 0
   525→  return {
   526→    content,
   527→    usage: {
   528→      promptTokens,
   529→      completionTokens,
   530→      totalTokens: json.usageMetadata?.totalTokenCount ?? promptTokens + completionTokens,
   531→    },
   532→  }
   533→}

   535→/** Ollama /api/chat — OpenAI-style but local. */
   536→async function callOllama(
   537→  baseUrl: string,
   538→  model: string,
   539→  messages: ChatMessage[],
   540→  opts: ChatOpts,
   541→): Promise<AdapterResult> {
   542→  const url = `${baseUrl.replace(/\/$/, '')}/api/chat`
   543→  const body: Record<string, unknown> = {
   544→    model,
   545→    messages,
   546→    stream: false,
   547→  }
   548→  const options: Record<string, unknown> = {}
   549→  if (typeof opts.maxTokens === 'number') options.num_predict = opts.maxTokens
   550→  if (typeof opts.temperature === 'number') options.temperature = opts.temperature
   551→  if (Object.keys(options).length) body.options = options
   552→
   553→  const res = await fetch(url, {
   554→    method: 'POST',
   555→    headers: { 'Content-Type': 'application/json' },
   556→    body: JSON.stringify(body),
   557→  })
   558→  if (!res.ok) {
   559→    const text = await res.text().catch(() => '')
   560→    throw new Error(`Ollama API ${res.status}: ${text.slice(0, 500)}`)
   561→  }
   562→  const json = (await res.json()) as {
   563→    message?: { content?: string }
   564→    prompt_eval_count?: number
   565→    eval_count?: number
   566→  }
   567→  return {
   568→    content: json.message?.content ?? '',
   569→    usage: {
   570→      promptTokens: json.prompt_eval_count ?? 0,
   571→      completionTokens: json.eval_count ?? 0,
   572→      totalTokens: (json.prompt_eval_count ?? 0) + (json.eval_count ?? 0),
   573→    },
   574→  }
   575→}

   577→/** ZAI fallback — always works in dev. Doesn't return token counts. */
   578→async function callZai(
   579→  model: string,
   580→  messages: ChatMessage[],
   581→  opts: ChatOpts,
   582→): Promise<AdapterResult> {
   583→  const zai = await ZAI.create()
   584→  // Don't pass `model` for apical:* models — ZAI uses its internal default.
   585→  const isApical = model.startsWith('apical-')
   586→  const completion = await zai.chat.completions.create({
   587→    ...(isApical ? {} : { model }),
   588→    messages,
   589→    ...(typeof opts.temperature === 'number' ? { temperature: opts.temperature } : {}),
   590→  })
   591→  const content = completion.choices?.[0]?.message?.content ?? ''
   592→  return { content, usage: estimateUsage(messages, content) }
   593→}

   595→// ---------------- Streaming adapters ----------------
   596→
   597→async function* streamOpenAI(
   598→  apiKey: string,
   599→  baseUrl: string,
   600→  model: string,
   601→  messages: ChatMessage[],
   602→  opts: ChatOpts,
   603→): AsyncGenerator<AdapterStreamChunk> {
   604→  const url = `${baseUrl.replace(/\/$/, '')}/chat/completions`
   605→  const body: Record<string, unknown> = {
   606→    model,
   607→    messages,
   608→    stream: true,
   609→    stream_options: { include_usage: true },
   610→  }
   611→  if (typeof opts.maxTokens === 'number') body.max_tokens = opts.maxTokens
   612→  if (typeof opts.temperature === 'number') body.temperature = opts.temperature
   613→
   614→  const res = await fetch(url, {
   615→    method: 'POST',
   616→    headers: {
   617→      'Content-Type': 'application/json',
   618→      Authorization: `Bearer ${apiKey}`,
   619→    },
   620→    body: JSON.stringify(body),
   621→  })
   622→  if (!res.ok || !res.body) {
   623→    const text = await res.text().catch(() => '')
   624→    throw new Error(`OpenAI stream ${res.status}: ${text.slice(0, 500)}`)
   625→  }
   626→
   627→  let promptTokens = 0
   628→  let completionTokens = 0
   629→  let totalTokens = 0
   630→  const reader = res.body.getReader()
   631→  const decoder = new TextDecoder()
   632→  let buf = ''
   633→  while (true) {
   634→    const { done, value } = await reader.read()
   635→    if (done) break
   636→    buf += decoder.decode(value, { stream: true })
   637→    const lines = buf.split('\n')
   638→    buf = lines.pop() ?? ''
   639→    for (const line of lines) {
   640→      const trimmed = line.trim()
   641→      if (!trimmed.startsWith('data:')) continue
   642→      const payload = trimmed.slice(5).trim()
   643→      if (!payload || payload === '[DONE]') continue
   644→      try {
   645→        const ev = JSON.parse(payload) as {
   646→          choices?: Array<{ delta?: { content?: string } }>
   647→          usage?: { prompt_tokens?: number; completion_tokens?: number; total_tokens?: number }
   648→        }
   649→        const delta = ev.choices?.[0]?.delta?.content
   650→        if (delta) yield { delta }
   651→        if (ev.usage) {
   652→          promptTokens = ev.usage.prompt_tokens ?? promptTokens
   653→          completionTokens = ev.usage.completion_tokens ?? completionTokens
   654→          totalTokens = ev.usage.total_tokens ?? totalTokens
   655→        }
   656→      } catch {
   657→        // Skip unparseable lines.
   658→      }
   659→    }
   660→  }
   661→  yield {
   662→    done: {
   663→      content: '',
   664→      usage: {
   665→        promptTokens,
   666→        completionTokens,
   667→        totalTokens: totalTokens || promptTokens + completionTokens,
   668→      },
   669→    },
   670→  }
   671→}

   673→async function* streamAnthropic(
   674→  apiKey: string,
   675→  baseUrl: string,
   676→  model: string,
   677→  messages: ChatMessage[],
   678→  opts: ChatOpts,
   679→): AsyncGenerator<AdapterStreamChunk> {
   680→  const url = `${baseUrl.replace(/\/$/, '')}/v1/messages`
   681→  const systemMsgs = messages.filter((m) => m.role === 'system')
   682→  const userMsgs = messages.filter((m) => m.role !== 'system')
   683→  const body: Record<string, unknown> = {
   684→    model,
   685→    messages: userMsgs,
   686→    max_tokens: opts.maxTokens ?? 1024,
   687→    stream: true,
   688→  }
   689→  if (systemMsgs.length) {
   690→    body.system = systemMsgs.map((m) => m.content).join('\n\n')
   691→  }
   692→  if (typeof opts.temperature === 'number') body.temperature = opts.temperature
   693→
   694→  const res = await fetch(url, {
   695→    method: 'POST',
   696→    headers: {
   697→      'Content-Type': 'application/json',
   698→      'x-api-key': apiKey,
   699→      'anthropic-version': '2023-06-01',
   700→    },
   701→    body: JSON.stringify(body),
   702→  })
   703→  if (!res.ok || !res.body) {
   704→    const text = await res.text().catch(() => '')
   705→    throw new Error(`Anthropic stream ${res.status}: ${text.slice(0, 500)}`)
   706→  }
   707→
   708→  let promptTokens = 0
   709→  let completionTokens = 0
   710→  const reader = res.body.getReader()
   711→  const decoder = new TextDecoder()
   712→  let buf = ''
   713→  while (true) {
   714→    const { done, value } = await reader.read()
   715→    if (done) break
   716→    buf += decoder.decode(value, { stream: true })
   717→    const lines = buf.split('\n')
   718→    buf = lines.pop() ?? ''
   719→    for (const line of lines) {
   720→      const trimmed = line.trim()
   721→      if (!trimmed.startsWith('data:')) continue
   722→      const payload = trimmed.slice(5).trim()
   723→      if (!payload) continue
   724→      try {
   725→        const ev = JSON.parse(payload) as {
   726→          type?: string
   727→          delta?: { text?: string }
   728→          message?: { usage?: { input_tokens?: number } }
   729→          usage?: { input_tokens?: number; output_tokens?: number }
   730→        }
   731→        if (ev.type === 'content_block_delta' && ev.delta?.text) {
   732→          yield { delta: ev.delta.text }
   733→        } else if (ev.type === 'message_start' && ev.message?.usage?.input_tokens) {
   734→          promptTokens = ev.message.usage.input_tokens
   735→        } else if (ev.type === 'message_delta' && ev.usage?.output_tokens) {
   736→          completionTokens = ev.usage.output_tokens
   737→        }
   738→      } catch {
   739→        // Skip.
   740→      }
   741→    }
   742→  }
   743→  yield {
   744→    done: {
   745→      content: '',
   746→      usage: {
   747→        promptTokens,
   748→        completionTokens,
   749→        totalTokens: promptTokens + completionTokens,
   750→      },
   751→    },
   752→  }

   815→    done: {

   826→async function* streamZai(
   827→  model: string,
   828→  messages: ChatMessage[],
   829→  opts: ChatOpts,
   830→): AsyncGenerator<AdapterStreamChunk> {
   831→  const zai = await ZAI.create()
   832→  // Don't pass `model` for apical:* models — ZAI doesn't recognize those names
   833→  // and the SDK uses its default. (Match the pattern in /api/agent/stream.)
   834→  const isApical = model.startsWith('apical-')
   835→  const completion = await zai.chat.completions.create({
   836→    ...(isApical ? {} : { model }),
   837→    messages,
   838→    stream: true,
   839→    ...(typeof opts.temperature === 'number' ? { temperature: opts.temperature } : {}),
   840→  })
   841→  let content = ''
   842→  for await (const chunk of completion as AsyncIterable<{
   843→    choices?: Array<{ delta?: { content?: string } }>
   844→  }>) {
   845→    const text = chunk.choices?.[0]?.delta?.content
   846→    if (text) {
   847→      content += text
   848→      yield { delta: text }
   849→    }
   850→  }
   851→  // If the stream yielded nothing, fall back to a non-streaming call so the
   852→  // caller still gets content + accurate usage.
   853→  if (!content) {
   854→    const fallback = await callZai(model, messages, opts)
   855→    yield { delta: fallback.content }
   856→    yield { done: { content: fallback.content, usage: fallback.usage } }
   857→    return
   858→  }
   859→  yield { done: { content, usage: estimateUsage(messages, content) } }
   860→}

   864→async function callAdapter(
   865→  resolved: ResolvedModel,
   866→  messages: ChatMessage[],
   867→  opts: ChatOpts,
   868→): Promise<AdapterResult> {
   869→  switch (resolved.adapter) {
   870→    case 'openai':
   871→      if (!resolved.apiKey) throw new Error('OpenAI API key not configured')
   872→      return callOpenAI(
   873→        resolved.apiKey,
   874→        resolved.baseUrl || 'https://api.openai.com/v1',
   875→        resolved.model.apiModelId,
   876→        messages,
   877→        opts,
   878→      )
   879→    case 'anthropic':
   880→      if (!resolved.apiKey) throw new Error('Anthropic API key not configured')
   881→      return callAnthropic(
   882→        resolved.apiKey,
   883→        resolved.baseUrl || 'https://api.anthropic.com',
   884→        resolved.model.apiModelId,
   885→        messages,
   886→        opts,
   887→      )
   888→    case 'google':
   889→      if (!resolved.apiKey) throw new Error('Google API key not configured')
   890→      return callGoogle(
   891→        resolved.apiKey,
   892→        resolved.baseUrl,
   893→        resolved.model.apiModelId,
   894→        messages,
   895→        opts,
   896→      )
   897→    case 'ollama':
   898→      return callOllama(
   899→        resolved.baseUrl || 'http://localhost:11434',
   900→        resolved.model.apiModelId,
   901→        messages,
   902→        opts,
   903→      )
   904→    case 'llamacpp':
   905→      // llama.cpp exposes an OpenAI-compatible endpoint.
   906→      return callOpenAI(
   907→        resolved.apiKey || 'no-key',
   908→        resolved.baseUrl || 'http://localhost:8080',
   909→        resolved.model.apiModelId,
   910→        messages,
   911→        opts,
   912→      )
   913→    case 'zai':
   914→    default:
   915→      return callZai(resolved.model.apiModelId, messages, opts)
   916→  }
   917→}

   972→export async function checkAllowance(userId: string): Promise<AllowanceStatus> {
   973→  const sub = await getOrCreateSubscription(userId)

   975→  const allowance = sub.tokenAllowanceMonthly || plan.tokenAllowanceMonthly
   976→  const used = sub.tokenUsedMonthly
   977→  const overage = Math.max(0, used - allowance)
   978→  const overrunEnabled = sub.overrunEnabled && plan.overrunAvailable
   979→  const allowed = !isOverAllowance(used, allowance) || overrunEnabled
   980→  return {
   981→    allowed,
   982→    used,
   983→    allowance,
   984→    overage,
   985→    overrunEnabled,
   986→    periodEnd: sub.currentPeriodEnd,
   987→  }

   990→export async function recordUsage(p: RecordUsageParams): Promise<void> {
   991→  const sub = await getOrCreateSubscription(p.userId)
   992→  const plan = getPlan(sub.plan)
   993→  const allowance = sub.tokenAllowanceMonthly || plan.tokenAllowanceMonthly
   994→
   995→  // Increment the period totals.
   996→  const newUsed = sub.tokenUsedMonthly + p.promptTokens + p.completionTokens
   997→  const nowOver = isOverAllowance(newUsed, allowance)
   998→  const overrunEnabled = sub.overrunEnabled && plan.overrunAvailable
   999→
  1000→  const update: {
  1001→    tokenUsedMonthly: number
  1002→    tokenOverageMonthly?: number
  1003→    overageAccruedCents?: number
  1004→  } = { tokenUsedMonthly: newUsed }
  1005→
  1006→  // If this call pushed the user over the allowance (or they were already
  1007→  // over) AND they've opted into overrun billing, accrue the overage.
  1008→  if (nowOver && overrunEnabled) {
  1009→    const tokensThisCall = p.promptTokens + p.completionTokens
  1010→    update.tokenOverageMonthly = sub.tokenOverageMonthly + tokensThisCall
  1011→    update.overageAccruedCents = sub.overageAccruedCents + p.costCents
  1012→  }
  1013→
  1014→  await db.subscription.update({ where: { id: sub.id }, data: update })
  1015→
  1016→  await db.tokenUsageRecord.create({
  1017→    data: {
  1018→      userId: p.userId,
  1019→      modelId: p.modelId,
  1020→      provider: p.provider,
  1021→      promptTokens: p.promptTokens,
  1022→      completionTokens: p.completionTokens,
  1023→      totalTokens: p.promptTokens + p.completionTokens,
  1024→      costCents: p.costCents,
  1025→      source: p.source,
  1026→      refId: p.refId,
  1027→    },
  1028→  })
  1029→}

  1068→export async function* chatStream(
  1069→  req: ChatRequest,
  1070→): AsyncGenerator<StreamEvent> {
  1071→  const resolved = await resolveModel(req.userId, req.modelId)
  1072→  if (!resolved) {
  1073→    throw new Error(
  1074→      `Model "${req.modelId}" not found or not configured for this user`,
  1075→    )
  1076→  }
  1077→
  1078→  const opts = { maxTokens: req.maxTokens, temperature: req.temperature }
  1079→  let content = ''
  1080→  let usage: ChatUsage = {
  1081→    promptTokens: 0,
  1082→    completionTokens: 0,
  1083→    totalTokens: 0,
  1084→  }
  1085→
  1086→  for await (const chunk of streamAdapter(resolved, req.messages, opts)) {
  1087→    if (chunk.delta) {
  1088→      content += chunk.delta
  1089→      yield { type: 'delta', content: chunk.delta }
  1090→    }
  1091→    if (chunk.done) {
  1092→      if (chunk.done.content) content = chunk.done.content
  1093→      usage = chunk.done.usage
  1094→    }
  1095→  }
  1096→
  1097→  // Fallback: if the adapter didn't yield a `done` with usage, estimate.
  1098→  if (usage.totalTokens === 0 && content.length > 0) {
  1099→    usage = estimateUsage(req.messages, content)
  1100→  } else if (usage.totalTokens === 0) {
  1101→    usage = estimateUsage(req.messages, '')
  1102→  }
  1103→
  1104→  const costCents = computeCost(resolved.model, usage)
  1105→
  1106→  await recordUsage({
  1107→    userId: req.userId,
  1108→    modelId: req.modelId,
  1109→    provider: resolved.model.provider,
  1110→    promptTokens: usage.promptTokens,
  1111→    completionTokens: usage.completionTokens,
  1112→    costCents,
  1113→    source: req.source ?? 'chat',
  1114→    refId: req.refId,
  1115→  })
  1116→
  1117→  yield { type: 'done', usage }
  1118→}
