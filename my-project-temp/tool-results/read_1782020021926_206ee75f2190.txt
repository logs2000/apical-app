   495→// Cheap way to detect the current plan client-side (reads the cached sub from
   496→// /api/billing/subscription if available). For the landing page we don't need
   497→// this to be live — just a best-effort label.
   498→const currentPlanId = 'free' // visitors on the landing are typically free
   499→
   500→function computePrice(
   501→  plan: (typeof PLAN_LIST)[number],
   502→  interval: 'monthly' | 'yearly',
   503→): { display: string; suffix?: string; sub?: string } {
   504→  if (plan.id === 'enterprise') return { display: 'Custom', sub: 'Volume pricing — talk to sales.' }
   505→  if (plan.id === 'free') return { display: '$0', suffix: '/mo', sub: 'Free forever. No credit card.' }
   506→  const perSeat = plan.id === 'team' ? ' /seat' : ''
   507→  if (interval === 'monthly') {
   508→    return {
   509→      display: `$${plan.priceMonthly}`,
   510→      suffix: `/mo${perSeat}`,
   511→      sub: plan.id === 'team' ? '5 seats included.' : 'Billed monthly.',
   512→    }
   513→  }
   514→  const perMonth = plan.priceYearly / 12
   515→  const monthsFree = 12 - Math.round(plan.priceYearly / plan.priceMonthly)
   516→  return {
   517→    display: `$${perMonth % 1 === 0 ? perMonth : perMonth.toFixed(2)}`,
   518→    suffix: `/mo${perSeat}`,
   519→    sub: `$${plan.priceYearly}/yr${perSeat} — ${monthsFree} months free`,
   520→  }
   521→}

   653→// Shared: Download button + dialog
   654→// ---------------------------------------------------------------------------
   655→
   656→function DownloadButton({
   657→  os,
   658→  variant = 'default',
   659→  size = 'default',
   660→  className,
   661→}: {
   662→  os: DetectedOS
   663→  variant?: 'default' | 'outline' | 'ghost'
   664→  size?: 'default' | 'sm' | 'lg'
   665→  className?: string
   666→}) {
   667→  const [open, setOpen] = React.useState(false)
   668→  const { toast } = useToast()
   669→
   670→  // The button tries the real download first. If the binary isn't uploaded
   671→  // (404 not_uploaded), it opens the dialog with the CLI fallback.
   672→  const handleClick = async () => {
   673→    if (os === 'other') {
   674→      setOpen(true)
   675→      return
   676→    }
   677→    // Hit the API. If it returns a file (200), the browser downloads it. If
   678→    // it returns 404 not_uploaded, we open the dialog.
   679→    try {
   680→      const res = await fetch(downloadUrl(os), { method: 'HEAD' })
   681→      if (res.ok) {
   682→        // Real binary exists — trigger the download.
   683→        window.location.href = downloadUrl(os)
   684→        return
   685→      }
   686→      // 404 or other — open the dialog with the fallback.
   687→      setOpen(true)
   688→    } catch {
   689→      setOpen(true)
   690→    }
   691→  }
   692→
   693→  return (
   694→    <>
   695→      <Button variant={variant} size={size} onClick={handleClick} className={className}>
   696→        <Download className="mr-1.5 h-4 w-4" />
   697→        {downloadButtonLabel(os)}
   698→      </Button>
   699→      <DownloadDialog open={open} onOpenChange={setOpen} os={os} onCopied={() => toast({ title: 'Copied' })} />
   700→    </>
   701→  )
   702→}
   703→
   704→function DownloadDialog({
   705→  open,
   706→  onOpenChange,
   707→  os,
   708→  onCopied,
   709→}: {
   710→  open: boolean
   711→  onOpenChange: (v: boolean) => void
   712→  os: DetectedOS
   713→  onCopied: () => void
   714→}) {
   715→  const cmd = installCommandFor(os)
   716→  const [copied, setCopied] = React.useState(false)
   717→
   718→  const copy = () => {
   719→    navigator.clipboard?.writeText(cmd).then(() => {
   720→      setCopied(true)
   721→      onCopied()
   722→      setTimeout(() => setCopied(false), 2000)
   723→    })
   724→  }
   725→
   726→  return (
   727→    <Dialog open={open} onOpenChange={onOpenChange}>
   728→      <DialogContent className="sm:max-w-md">
   729→        <DialogHeader>
   730→          <DialogTitle>Install Apical for {osLabel(os)}</DialogTitle>
   731→          <DialogDescription>
   732→            {os === 'other'
   733→              ? 'Pick your platform below.'
   734→              : 'The desktop app is the fastest way to get started. If the download above didn\'t start, use the command below.'}
   735→          </DialogDescription>
   736→        </DialogHeader>
   737→
   738→        {os !== 'other' && (
   739→          <div className="space-y-3">
   740→            <a
   741→              href={downloadUrl(os)}
   742→              className="flex items-center justify-center gap-2 rounded-lg bg-primary px-4 py-2.5 text-sm font-medium text-primary-foreground transition-opacity hover:opacity-90"
   743→            >
   744→              <Download className="h-4 w-4" /> Download {osLabel(os)} app
   745→            </a>
   746→
   747→            <div className="relative">
   748→              <div className="flex items-center gap-2 rounded-lg border border-border bg-muted/50 p-2.5 pl-3">
   749→                <Terminal className="h-3.5 w-3.5 shrink-0 text-muted-foreground" />
   750→                <code className="flex-1 truncate font-mono text-xs">{cmd}</code>
   751→                <Button variant="ghost" size="icon" className="h-7 w-7 shrink-0" onClick={copy}>
   752→                  {copied ? <Check className="h-3.5 w-3.5 text-primary" /> : <Copy className="h-3.5 w-3.5" />}
   753→                </Button>
   754→              </div>
   755→              <p className="mt-2 text-[11px] text-muted-foreground">
   756→                Or install from the command line.
   757→              </p>
   758→            </div>
   759→          </div>
   760→        )}
   761→
   762→        {os === 'other' && (
   763→          <div className="space-y-2">
   764→            {(['mac', 'windows', 'linux'] as DetectedOS[]).map((o) => (
   765→              <a
   766→                key={o}
   767→                href={downloadUrl(o)}
   768→                className="flex items-center justify-between rounded-lg border border-border p-3 text-sm hover:bg-accent/50"
   769→              >
   770→                <span>{osLabel(o)}</span>
   771→                <Download className="h-4 w-4 text-muted-foreground" />
   772→              </a>
   773→            ))}
   774→          </div>
   775→        )}
   776→
   777→        <DialogFooter className="flex-col gap-2 sm:flex-col sm:space-x-0">
   778→          <Button variant="outline" className="w-full" asChild>
   779→            <a href="/docs">Build from source</a>
   780→          </Button>
   781→        </DialogFooter>
   782→      </DialogContent>
   783→    </Dialog>
   784→  )
   785→}
