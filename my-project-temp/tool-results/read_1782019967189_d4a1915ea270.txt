     1→// Apical agent tools — the registry of tools the autonomous agent can call
     2→// during a reasoning loop. Each tool has a strict JSON schema + an executor
     3→// that returns a structured observation.
     4→//
     5→// Tools are the agent's hands. The engine (agent-engine.ts) is the brain.
     6→//
     7→// Security model:
     8→//   - web_search, web_read, http_request: network egress (the agent can reach
     9→//     the web to research).
    10→//   - code_eval: a sandboxed JS eval (no filesystem, no require, no process).
    11→//     Used for computations, data transformation, regex extraction.
    12→//   - cli_run: DISABLED by default (requires an explicit enable flag). When
    13→//     enabled, routes through the desktop bridge so commands run on the user's
    14→//     machine, not the server.
    15→//   - data_table_*: CRUD on the user's built-in DataTables.
    16→//   - workflow_propose: lets the agent emit a workflow draft (not deploy).
    17→//   - integration_list: lets the agent see what connections are available.
    18→

    32→export interface ToolResult {

    36→  /** Optional display hints for the UI. */
    37→  display?: {
    38→    title: string

    40→    kind?: 'search' | 'http' | 'code' | 'cli' | 'data' | 'workflow' | 'info'
    41→  }

    44→export interface ToolContext {
    45→  userId: string
    46→  /** Whether CLI execution is allowed (routes through the desktop bridge). */
    47→  allowCli: boolean
    48→  /** Max HTTP fetch size (bytes). */
    49→  maxFetchBytes: number
    50→  /** The agent's accumulated workflow proposal (mutated by workflow_propose). */
    51→  proposedWorkflow?: WorkflowJSON
    52→  /** A growing list of research findings (mutated by web_read/http_request). */
    53→  findings?: Array<{ source: string; url: string; type: string; description: string }>

    56→export interface ToolDef {
    57→  name: string
    58→  description: string
    59→  /** JSON-schema-ish input shape for the LLM. */
    60→  inputSchema: Record<string, { type: string; description: string; required?: boolean }>
    61→  /** Run the tool. Must not throw — return { ok: false, error } on failure. */
    62→  run: (input: Record<string, unknown>, ctx: ToolContext) => Promise<ToolResult>
    63→}

    65→// ---------------- Helpers ----------------
    66→
    67→function asString(v: unknown, max = 100_000): string {
    68→  if (typeof v !== 'string') return ''
    69→  return v.slice(0, max)
    70→}
    71→
    72→function asNumber(v: unknown, def = 0): number {
    73→  const n = Number(v)
    74→  return Number.isFinite(n) ? n : def
    75→}
    76→
    77→function truncate(s: string, max: number): string {
    78→  if (s.length <= max) return s
    79→  return s.slice(0, max) + `\n…[truncated, ${s.length - max} more chars]`
    80→}

   122→          summary: `${rows.length} results for "${query}"`,

   125→      }
   126→    } catch (e) {

   129→  },

   132→// 2. web_read — fetch + extract the main content of a URL.
   133→const webRead: ToolDef = {
   134→  name: 'web_read',
   135→  description:
   136→    'Fetch a web page and extract its main text content. Use this to read a page found via web_search, inspect an API response, or read documentation.',
   137→  inputSchema: {
   138→    url: { type: 'string', description: 'The URL to fetch.', required: true },
   139→    maxChars: { type: 'number', description: 'Max content length (default 8000).' },
   140→  },
   141→  async run(input, ctx) {
   142→    const url = asString(input.url, 2000)
   143→    if (!url || !/^https?:\/\//.test(url))
   144→      return { ok: false, output: null, error: 'valid http(s) url is required' }
   145→    const maxChars = Math.min(50_000, Math.max(500, asNumber(input.maxChars, 8000)))
   146→
   147→    // Prefer a raw fetch + tag strip — it's reliable + doesn't depend on the
   148→    // SDK's web_reader function (which may not be available). Try the SDK
   149→    // reader only if the raw fetch fails (e.g. behind a paywall the SDK can
   150→    // bypass).
   151→    let title = ''
   152→    let content = ''
   153→    let publishedTime: string | undefined
   154→    let usedMethod = 'fetch'
   155→
   156→    try {
   157→      const r = await fetch(url, {
   158→        signal: AbortSignal.timeout(12_000),
   159→        headers: { 'User-Agent': 'Apical-Research-Bot/1.0', Accept: 'text/html,application/json,text/plain,*/*' },
   160→      })
   161→      if (r.ok) {
   162→        const ct = r.headers.get('content-type') || ''
   163→        const raw = await r.text()
   164→        if (ct.includes('json') || raw.trim().startsWith('{') || raw.trim().startsWith('[')) {
   165→          // JSON response — return it parsed.
   166→          try {
   167→            content = JSON.stringify(JSON.parse(raw), null, 2)
   168→          } catch {
   169→            content = raw
   170→          }
   171→        } else {
   172→          // HTML — strip tags + scripts.
   173→          const titleMatch = raw.match(/<title[^>]*>([^<]*)<\/title>/i)
   174→          if (titleMatch) title = titleMatch[1].trim().slice(0, 300)
   175→          content = raw
   176→            .replace(/<script[\s\S]*?<\/script>/gi, '')
   177→            .replace(/<style[\s\S]*?<\/style>/gi, '')
   178→            .replace(/<[^>]+>/g, ' ')
   179→            .replace(/&[a-z]+;/gi, ' ')
   180→            .replace(/\s+/g, ' ')
   181→            .trim()
   182→        }
   183→      } else {
   184→        // Raw fetch failed — try the SDK reader as a fallback.
   185→        usedMethod = 'sdk'
   186→        try {
   187→          const zai = await getZai()
   188→          const res = (await zai.functions.invoke('web_reader', { url })) as { title?: string; content?: string; publishedTime?: string }
   189→          title = asString(res.title || '', 300)
   190→          content = asString(res.content || '', maxChars)
   191→          publishedTime = res.publishedTime
   192→        } catch (e2) {
   193→          return { ok: false, output: null, error: `HTTP ${r.status}` }
   194→        }
   195→      }
   196→    } catch (e) {
   197→      // Raw fetch threw (network error) — try the SDK reader.
   198→      usedMethod = 'sdk'
   199→      try {
   200→        const zai = await getZai()
   201→        const res = (await zai.functions.invoke('web_reader', { url })) as { title?: string; content?: string; publishedTime?: string }
   202→        title = asString(res.title || '', 300)
   203→        content = asString(res.content || '', maxChars)
   204→        publishedTime = res.publishedTime
   205→      } catch (e2) {
   206→        return { ok: false, output: null, error: (e as Error).message }
   207→      }
   208→    }
   209→
   210→    content = truncate(content, maxChars)
   211→    // Record a finding for the workflow proposal.
   212→    ctx.findings?.push({
   213→      source: title || url,
   214→      url,
   215→      type: 'website',
   216→      description: truncate(content, 200),
   217→    })
   218→    const record = { url, title, content, publishedTime, method: usedMethod }
   219→    return {
   220→      ok: true,
   221→      output: record,
   222→      display: { title: `Read ${url}`, summary: title || 'read', kind: 'http' },
   223→    }
   224→  },
   225→}

   227→// 3. http_request — a raw HTTP call (for APIs the agent discovers).
   228→const httpRequest: ToolDef = {
   229→  name: 'http_request',
   230→  description:
   231→    'Make a raw HTTP request to any URL. Use this to call an API endpoint you discovered (e.g. GET https://api.example.com/v1/items). Returns the status, headers, and body (truncated).',
   232→  inputSchema: {
   233→    url: { type: 'string', description: 'The URL.', required: true },
   234→    method: { type: 'string', description: 'HTTP method (default GET).' },
   235→    headers: { type: 'object', description: 'Request headers (key-value).' },
   236→    body: { type: 'string', description: 'Request body (for POST/PUT).' },
   237→  },
   238→  async run(input, ctx) {
   239→    const url = asString(input.url, 2000)
   240→    if (!url || !/^https?:\/\//.test(url))
   241→      return { ok: false, output: null, error: 'valid http(s) url is required' }
   242→    const method = (asString(input.method, 10) || 'GET').toUpperCase()
   243→    const headers = (input.headers as Record<string, string>) ?? {}
   244→    const body = asString(input.body, 100_000)
   245→    try {
   246→      const r = await fetch(url, {
   247→        method,
   248→        headers,
   249→        body: ['GET', 'HEAD'].includes(method) ? undefined : body,
   250→        signal: AbortSignal.timeout(15_000),
   251→      })
   252→      const text = await r.text()
   253→      const ct = r.headers.get('content-type') || ''
   254→      // Try to parse JSON; else return text (truncated).
   255→      let parsed: unknown = text
   256→      if (ct.includes('json') || text.trim().startsWith('{') || text.trim().startsWith('[')) {
   257→        try {
   258→          parsed = JSON.parse(text)
   259→        } catch {
   260→          /* keep text */
   261→        }
   262→      }
   263→      const out = {
   264→        status: r.status,
   265→        ok: r.ok,
   266→        headers: Object.fromEntries(r.headers.entries()),
   267→        body: typeof parsed === 'string' ? truncate(parsed, ctx.maxFetchBytes) : parsed,
   268→      }
   269→      return {
   270→        ok: r.ok,
   271→        output: out,
   272→        display: { title: `${method} ${url}`, summary: `HTTP ${r.status}`, kind: 'http' },
   273→      }
   274→    } catch (e) {
   275→      return { ok: false, output: null, error: (e as Error).message }
   276→    }
   277→  },
   278→}

   280→// 4. code_eval — sandboxed JS for computations + data transformation.
   281→//    NO filesystem, NO require, NO process, NO fetch (the agent uses
   282→//    http_request for network). Pure computation only.
   283→const codeEval: ToolDef = {
   284→  name: 'code_eval',
   285→  description:
   286→    'Run a snippet of JavaScript to compute, transform, or analyze data. Use this to parse JSON, run regex extraction, do math, filter/sort arrays, or format output. No filesystem, no network, no require — pure computation only. The last expression is the result.',
   287→  inputSchema: {
   288→    code: { type: 'string', description: 'JavaScript to evaluate. The last expression is returned.', required: true },
   289→    data: { type: 'string', description: 'Optional JSON string to parse into a `data` variable.' },
   290→  },
   291→  async run(input, _ctx) {
   292→    const code = asString(input.code, 20_000)
   293→    if (!code) return { ok: false, output: null, error: 'code is required' }
   294→    let data: unknown = undefined
   295→    if (input.data) {
   296→      try {
   297→        data = JSON.parse(asString(input.data, 100_000))
   298→      } catch {
   299→        return { ok: false, output: null, error: 'data is not valid JSON' }
   300→      }
   301→    }
   302→    try {
   303→      // Sandbox: wrap in a function with no access to globals. We provide a
   304→      // minimal `data` binding + JSON + Math + standard built-ins via an
   305→      // indirect eval in a sealed scope.
   306→      const fn = new Function(
   307→        'data',
   308→        '"use strict";\n' +
   309→          'const Math_ = Math, JSON_ = JSON, Date_ = Date, Array_ = Array, Object_ = Object, String_ = String, Number_ = Number, RegExp_ = RegExp;\n' +
   310→          'return (function(){\n' +
   311→          code +
   312→          '\n})();',
   313→      )
   314→      const result = fn(data)
   315→      return {
   316→        ok: true,
   317→        output: { result: typeof result === 'string' ? truncate(result, 10_000) : result },
   318→        display: { title: 'Ran code', summary: 'evaluated JS', kind: 'code' },
   319→      }
   320→    } catch (e) {
   321→      return { ok: false, output: null, error: (e as Error).message }
   322→    }
   323→  },
   324→}

   326→// 5. cli_run — run a command on the user's desktop (via the desktop bridge).
   327→//    Disabled unless ctx.allowCli is true.
   328→const cliRun: ToolDef = {
   329→  name: 'cli_run',
   330→  description:
   331→    'Run a shell command on the user\'s desktop (requires the desktop app to be connected). Use this to inspect local files, run network tools (nmap, arp), query local databases, or run scripts. DISABLED unless the user has enabled CLI access.',
   332→  inputSchema: {
   333→    command: { type: 'string', description: 'The command to run.', required: true },
   334→    args: { type: 'array', description: 'Command arguments.', items: { type: 'string' } },
   335→    cwd: { type: 'string', description: 'Working directory.' },
   336→    timeoutMs: { type: 'number', description: 'Timeout in ms (default 15000, max 30000).' },
   337→  },
   338→  async run(input, ctx) {
   339→    if (!ctx.allowCli)
   340→      return {
   341→        ok: false,
   342→        output: null,
   343→        error: 'CLI access is disabled. The user must enable it in Settings → Desktop.',
   344→      }
   345→    const command = asString(input.command, 2000)
   346→    if (!command) return { ok: false, output: null, error: 'command is required' }
   347→    const args = Array.isArray(input.args) ? input.args.map(String) : []
   348→    const cwd = asString(input.cwd, 1000) || undefined
   349→    const timeoutMs = Math.min(30_000, Math.max(1000, asNumber(input.timeoutMs, 15_000)))
   350→    try {
   351→      // Route through the desktop bridge. We need a session id — for the
   352→      // autonomous agent we use the user's first online desktop session.
   353→      const sessions = await db.desktopSession.findMany({
   354→        where: { userId: ctx.userId, status: 'online' },
   355→        take: 1,
   356→      })
   357→      if (sessions.length === 0)
   358→        return { ok: false, output: null, error: 'No online desktop session. Connect the desktop app.' }
   359→      const sessionId = sessions[0].id
   360→      const r = await fetch('http://localhost:3005/invoke', {
   361→        method: 'POST',
   362→        headers: { 'Content-Type': 'application/json' },
   363→        body: JSON.stringify({
   364→          sessionId,
   365→          tool: 'desktop.cli.run',
   366→          args: { command, args, cwd, timeoutMs },
   367→          timeoutMs: timeoutMs + 2000,
   368→        }),
   369→      })
   370→      const data = (await r.json()) as { ok: boolean; result?: unknown; error?: string }
   371→      return {
   372→        ok: data.ok,
   373→        output: data.result ?? null,
   374→        error: data.error,
   375→        display: { title: `$ ${command}`, summary: data.ok ? 'ran' : 'failed', kind: 'cli' },
   376→      }
   377→    } catch (e) {
   378→      return { ok: false, output: null, error: (e as Error).message }
   379→    }
   380→  },
   381→}

   456→// 6c. mcp_call_tool — call a tool on a connected MCP server.
   457→//     The agent uses this to actually USE a capability it discovered via
   458→//     mcp_list_servers (e.g. read a file via the filesystem MCP, query a
   459→//     database via the postgres MCP, etc.).
   460→const mcpCallTool: ToolDef = {
   461→  name: 'mcp_call_tool',
   462→  description:
   463→    'Call a tool on a connected MCP server. First use mcp_list_servers to discover the server id + tool name, then call this with the server id, the tool name, and the args. Returns the tool\'s raw result.',
   464→  inputSchema: {
   465→    serverId: { type: 'string', description: 'The MCP server id (from mcp_list_servers).', required: true },
   466→    tool: { type: 'string', description: 'The tool name to call on that server.', required: true },
   467→    args: { type: 'object', description: 'Arguments to pass to the tool (key-value).' },
   468→  },
   469→  async run(input, ctx) {
   470→    const serverId = asString(input.serverId, 100)
   471→    const toolName = asString(input.tool, 200)
   472→    if (!serverId || !toolName)
   473→      return { ok: false, output: null, error: 'serverId and tool are required' }
   474→    const args = (input.args as Record<string, unknown>) ?? {}
   475→    try {
   476→      // Look up the MCP integration by id. The Integration model has no
   477→      // userId column (integrations are shared), so we query by id + kind.
   478→      const row = await db.integration.findFirst({
   479→        where: { id: serverId, kind: 'mcp' },
   480→      })
   481→      if (!row) return { ok: false, output: null, error: 'MCP server not found' }
   482→      const cfg = parseConfig<{ mcp?: McpServerConfig }>(row.config, {})
   483→      if (!cfg.mcp) return { ok: false, output: null, error: 'MCP server config missing' }
   484→      const result = await callMcpTool(cfg.mcp, toolName, args)
   485→      // callMcpTool returns { error } on failure, or the raw result on success.
   486→      const errObj = result as { error?: string }
   487→      if (errObj && typeof errObj.error === 'string') {
   488→        return {
   489→          ok: false,
   490→          output: null,
   491→          error: errObj.error,
   492→          display: { title: `${row.name}.${toolName}`, summary: 'failed', kind: 'info' },
   493→        }
   494→      }
   495→      return {
   496→        ok: true,
   497→        output: result,
   498→        display: { title: `${row.name}.${toolName}`, summary: 'called', kind: 'info' },
   499→      }
   500→    } catch (e) {
   501→      return { ok: false, output: null, error: (e as Error).message }
   502→    }
   503→  },
   504→}

   506→// 7. data_table_create — create a structured data table for storing results.

   622→// 10. workflow_propose — emit/update the workflow draft.
   623→const workflowPropose: ToolDef = {
   624→  name: 'workflow_propose',
   625→  description:
   626→    'Create or update the proposed workflow for this task. Call this once you have enough information to design the automation. The user will review + approve it before it runs.',
   627→  inputSchema: {
   628→    name: { type: 'string', description: 'Agent name (e.g. "LeadFinder").', required: true },
   629→    description: { type: 'string', description: 'One-line description of what the agent does.', required: true },
   630→    steps: {
   631→      type: 'array',
   632→      description: 'Workflow steps. Each step has a kind: "tool" | "reason" | "gate".',
   633→      items: { type: 'object' },
   634→      required: true,
   635→    },
   636→    schedule: { type: 'string', description: 'How often it should run (e.g. "Daily at 9am", "Every 30 min", "Weekly").' },
   637→  },
   638→  async run(input, ctx) {
   639→    const name = asString(input.name, 200)
   640→    const description = asString(input.description, 1000)
   641→    const steps = Array.isArray(input.steps) ? input.steps : []
   642→    if (!name || !description || steps.length === 0)
   643→      return { ok: false, output: null, error: 'name, description, and steps are required' }
   644→    // Normalize steps into the WorkflowStep shape.
   645→    const normalized = steps.map((s, i) => {
   646→      const step = s as Record<string, unknown>
   647→      const kind = step.kind === 'reason' || step.kind === 'gate' ? step.kind : 'tool'
   648→      const id = typeof step.id === 'string' && step.id ? step.id : `s${i + 1}`
   649→      const label = typeof step.label === 'string' && step.label ? step.label : kind === 'reason' ? 'Reason' : kind === 'gate' ? 'Approve' : 'Tool'
   650→      const out: Record<string, unknown> = { id, kind, label }
   651→      if (kind === 'tool') {
   652→        if (typeof step.tool === 'string') out.tool = step.tool
   653→        if (step.inputs && typeof step.inputs === 'object') out.inputs = step.inputs
   654→      } else if (kind === 'reason') {
   655→        if (typeof step.prompt === 'string') out.prompt = step.prompt
   656→        if (Array.isArray(step.allowedTools)) out.allowedTools = step.allowedTools
   657→      } else if (kind === 'gate') {
   658→        if (typeof step.gateMessage === 'string') out.gateMessage = step.gateMessage
   659→      }
   660→      if (typeof step.note === 'string') out.note = step.note
   661→      return out
   662→    })
   663→    const wf: WorkflowJSON = { version: 1, steps: normalized as never }
   664→    ctx.proposedWorkflow = wf
   665→    return {
   666→      ok: true,
   667→      output: { name, description, stepCount: normalized.length, schedule: asString(input.schedule, 200) },
   668→      display: {
   669→        title: `Proposed agent: ${name}`,
   670→        summary: `${normalized.length} steps`,
   671→        kind: 'workflow',
   672→      },
   673→    }
   674→  },
   675→}
