    84→// `EVOCATIVE_NAMES` is imported from @/lib/apical — single source of truth.
    85→// (Local set used for fast lookup in `isLikelyEvocativeName`.)
    86→const EVOCATIVE_NAME_SET = new Set<string>(EVOCATIVE_NAMES)

    88→function isLikelyEvocativeName(s: string): boolean {
    89→  const n = s.trim()
    90→  if (!n) return false
    91→  // Reject human-y names (Pat, Sam, etc.) and thing-words.
    92→  if (EVOCATIVE_NAME_SET.has(n)) return true
    93→  // Single capitalized word, 3-12 chars, not a thing-word.
    94→  if (!/^[A-Z][a-z]{2,11}$/.test(n)) return false
    95→  if (/(pdf|sorter|scanner|file|invoice|expense|workflow|tool|auto|bot|agent|pdf|doc|email|api)$/i.test(n)) return false
    96→  return true
    97→}

    99→/**
   100→ * Pick a name for a new agent using the user's preferred style.
   101→ * The LLM no longer decides the name — it just suggests one, and we override
   102→ * with `generateAgentName(style, jobDescription, existingNames)`.
   103→ *
   104→ * For evocative style: if the LLM suggested a valid evocative name AND it
   105→ * isn't already in use, we keep it (so the model can pick a fitting name).
   106→ * Otherwise we pick an unused one from the EVOCATIVE_NAMES list.
   107→ *
   108→ * For descriptive style: always derive from the job description (the LLM's
   109→ * suggestion is ignored — the user wants descriptive names).
   110→ */
   111→async function pickAgentName(
   112→  style: AgentNameStyle,
   113→  suggested: unknown,
   114→  jobDescription: string,
   115→  existingNames: string[],
   116→): Promise<string> {
   117→  if (style === 'descriptive') {
   118→    return generateAgentName('descriptive', jobDescription, existingNames)
   119→  }
   120→  // For evocative style, try the LLM to generate a fitting name.
   121→  try {
   122→    const zai = await ZAI.create()
   123→    const prompt = `Generate a single short, evocative, non-human name for an AI agent that does this job: "${jobDescription}".
   124→
   125→Rules:
   126→- 4-6 letters, easy to pronounce, memorable (like Nomi, Vexa, Kiro, Sova, Lumo, Talo)
   127→- NOT a common human name (not John, Sarah, Alex)
   128→- NOT a real word (not Scan, Sort, Mail)
   129→- Just the name, no explanation
   130→
   131→Existing names to avoid: ${existingNames.join(', ') || 'none'}
   132→
   133→Respond with ONLY the name, nothing else.`
   134→
   135→    const completion = await zai.chat.completions.create({
   136→      messages: [
   137→        { role: 'assistant', content: 'You generate short, evocative names for AI agents. Respond with a single word.' },
   138→        { role: 'user', content: prompt },
   139→      ],
   140→      thinking: { type: 'disabled' },
   141→    })
   142→    const name = (completion.choices[0]?.message?.content || '').trim().split(/\s+/)[0]
   143→    if (name && name.length >= 2 && name.length <= 8 && /^[a-zA-Z]+$/.test(name)) {
   144→      const taken = existingNames.some((n) => n.toLowerCase() === name.toLowerCase())
   145→      if (!taken) return name.charAt(0).toUpperCase() + name.slice(1)
   146→    }
   147→  } catch {
   148→    // Fall through to fallback.
   149→  }
   150→  // Fallback: prefer the LLM's original suggestion if valid.
   151→  if (typeof suggested === 'string' && suggested.trim()) {
   152→    const first = suggested.trim().split(/\s+/)[0]
   153→    const taken = existingNames.some((n) => n.trim().toLowerCase() === first.toLowerCase())
   154→    if (isLikelyEvocativeName(first) && !taken) return first
   155→  }
   156→  return generateAgentName('evocative', jobDescription, existingNames)
   157→}

   159→// Legacy alias — kept so the fallback path doesn't need to know about styles.
   160→function pickEvocativeName(suggested: unknown): string {
   161→  if (typeof suggested === 'string' && suggested.trim()) {
   162→    const first = suggested.trim().split(/\s+/)[0]
   163→    if (isLikelyEvocativeName(first)) return first
   164→  }
   165→  return EVOCATIVE_NAMES[Math.floor(Math.random() * EVOCATIVE_NAMES.length)]
   166→}

   179→/** Render an agent's steps compactly so the model can suggest edits. */
   180→function renderStepsBrief(steps: WorkflowStep[]): string {
   181→  return steps
   182→    .map((s) => {
   183→      const parts = [`  - ${s.id} [${s.kind}] ${s.label}`]
   184→      if (s.tool) parts.push(`      tool: ${s.tool}`)
   185→      if (s.prompt) parts.push(`      prompt: ${s.prompt}`)
   186→      if (s.gateMessage) parts.push(`      gate: ${s.gateMessage}`)
   187→      if (s.allowedTools && s.allowedTools.length)
   188→        parts.push(`      allowedTools: ${s.allowedTools.join(', ')}`)
   189→      if (s.hardened) parts.push(`      (hardened — rule applied, no AI)`)
   190→      return parts.join('\n')
   191→    })
   192→    .join('\n')
   193→}

   201→/** Normalize step list defensively — sequential ids, valid kinds, sane defaults. */
   202→function normalizeProposalSteps(raw: unknown): WorkflowStep[] {
   203→  const arr = parseStepsFromJson(raw)
   204→  const steps: WorkflowStep[] = []
   205→  arr.forEach((s, i) => {
   206→    const kind = s.kind === 'reason' || s.kind === 'gate' ? s.kind : 'tool'
   207→    const id = typeof s.id === 'string' && s.id ? s.id : `s${i + 1}`
   208→    const label =
   209→      typeof s.label === 'string' && s.label
   210→        ? s.label
   211→        : kind === 'reason'
   212→          ? 'Reason'
   213→          : kind === 'gate'
   214→            ? 'Approve'
   215→            : 'Run tool'
   216→    const out: WorkflowStep = { id, kind, label }
   217→    if (kind === 'tool') {
   218→      if (typeof s.tool === 'string') out.tool = s.tool
   219→      if (s.inputs && typeof s.inputs === 'object') {
   220→        out.inputs = s.inputs as Record<string, unknown>
   221→      }
   222→    } else if (kind === 'reason') {
   223→      if (typeof s.prompt === 'string') out.prompt = s.prompt
   224→      if (Array.isArray(s.allowedTools)) {
   225→        out.allowedTools = s.allowedTools.filter(
   226→          (t) => typeof t === 'string',
   227→        ) as string[]
   228→      }
   229→      if (s.outputShape && typeof s.outputShape === 'object') {
   230→        out.outputShape = s.outputShape as Record<string, string>
   231→      }
   232→      if (typeof s.confidenceThreshold === 'number') {
   233→        out.confidenceThreshold = s.confidenceThreshold
   234→      }
   235→    } else if (kind === 'gate') {
   236→      if (typeof s.gateMessage === 'string') out.gateMessage = s.gateMessage
   237→    }
   238→    if (typeof s.note === 'string') out.note = s.note
   239→    steps.push(out)
   240→  })
   241→  return steps.map((s, i) => ({ ...s, id: `s${i + 1}` }))
   242→}

   244→function coerceClarification(raw: unknown): ClarificationQuestion | undefined {
   245→  if (!raw || typeof raw !== 'object') return undefined
   246→  const r = raw as Record<string, unknown>
   247→  const question =
   248→    typeof r.question === 'string' && r.question.trim() ? r.question.trim() : ''
   249→  const optsRaw = Array.isArray(r.options) ? r.options : []
   250→  const options: ClarificationOption[] = optsRaw
   251→    .filter((o): o is Record<string, unknown> => !!o && typeof o === 'object')
   252→    .slice(0, 4)
   253→    .map((o, i) => {
   254→      const key =
   255→        typeof o.key === 'string' && o.key ? o.key : String.fromCharCode(65 + i)
   256→      const label =
   257→        typeof o.label === 'string' && o.label ? o.label : `Option ${i + 1}`
   258→      const description =
   259→        typeof o.description === 'string' ? o.description : undefined
   260→      return { key, label, description }
   261→    })
   262→    .filter((o) => o.label)
   263→  if (!question || options.length < 2) return undefined
   264→  return {
   265→    id: typeof r.id === 'string' && r.id ? r.id : 'clarify',
   266→    question,
   267→    options,
   268→    multiple: r.multiple === true,
   269→  }
   270→}

   272→function coerceApiDiscovery(raw: unknown): ApiDiscoveryCandidate[] | undefined {
   273→  if (!Array.isArray(raw)) return undefined
   274→  const out: ApiDiscoveryCandidate[] = []
   275→  raw.forEach((c, i) => {
   276→    if (!c || typeof c !== 'object') return
   277→    const r = c as Record<string, unknown>
   278→    const service =
   279→      typeof r.service === 'string' && r.service.trim() ? r.service.trim() : ''
   280→    if (!service) return
   281→    const kind: IntegrationKind =
   282→      r.kind === 'mcp' || r.kind === 'http' || r.kind === 'api'
   283→        ? (r.kind as IntegrationKind)
   284→        : 'api'
   285→    const toolsRaw = Array.isArray(r.tools) ? r.tools : []
   286→    const tools = toolsRaw
   287→      .filter(
   288→        (t): t is Record<string, unknown> => !!t && typeof t === 'object',
   289→      )
   290→      .slice(0, 6)
   291→      .map((t, j) => {
   292→        const id =
   293→          typeof t.id === 'string' && t.id
   294→            ? t.id
   295→            : `${service.toLowerCase().replace(/[^a-z0-9]+/g, '')}.tool${j + 1}`
   296→        const name =
   297→          typeof t.name === 'string' && t.name ? t.name : `Tool ${j + 1}`
   298→        const description =
   299→          typeof t.description === 'string' ? t.description : ''
   300→        return { id, name, description }
   301→      })
   302→    const fieldsRaw = Array.isArray(r.credentialFields) ? r.credentialFields : []
   303→    const credentialFields = fieldsRaw
   304→      .filter(
   305→        (f): f is Record<string, unknown> => !!f && typeof f === 'object',
   306→      )
   307→      .slice(0, 5)
   308→      .map((f, j) => {
   309→        const key =
   310→          typeof f.key === 'string' && f.key ? f.key : `field_${j + 1}`
   311→        const label =
   312→          typeof f.label === 'string' && f.label ? f.label : `Field ${j + 1}`
   313→        const type: 'apikey' | 'oauth' | 'mcp_token' =
   314→          f.type === 'oauth' || f.type === 'mcp_token' || f.type === 'apikey'
   315→            ? (f.type as 'apikey' | 'oauth' | 'mcp_token')
   316→            : 'apikey'
   317→        const placeholder =
   318→          typeof f.placeholder === 'string' ? f.placeholder : undefined
   319→        const required = f.required !== false
   320→        return { key, label, type, placeholder, required }
   321→      })
   322→    if (credentialFields.length === 0) {
   323→      // Default: a single API key field.
   324→      credentialFields.push({
   325→        key: 'api_key',
   326→        label: 'API key',
   327→        type: 'apikey',
   328→        placeholder: 'Paste your API key here',
   329→        required: true,
   330→      })
   331→    }
   332→    out.push({
   333→      id: typeof r.id === 'string' && r.id ? r.id : `cand_${i + 1}`,
   334→      service,
   335→      kind,
   336→      specUrl: typeof r.specUrl === 'string' ? r.specUrl : undefined,
   337→      url: typeof r.url === 'string' ? r.url : undefined,
   338→      description:
   339→        typeof r.description === 'string' && r.description
   340→          ? r.description
   341→          : `Connect ${service} so your agents can use it.`,
   342→      tools,
   343→      credentialFields,
   344→    })
   345→  })
   346→  if (out.length === 0) return undefined
   347→  return out
   348→}

   350→function coerceSuggestions(raw: unknown): Suggestion[] | undefined {
   351→  if (!Array.isArray(raw)) return undefined
   352→  const out: Suggestion[] = []
   353→  raw.forEach((s) => {
   354→    if (!s || typeof s !== 'object') return
   355→    const r = s as Record<string, unknown>
   356→    const title =
   357→      typeof r.title === 'string' && r.title.trim() ? r.title.trim() : ''
   358→    const prompt =
   359→      typeof r.prompt === 'string' && r.prompt.trim() ? r.prompt.trim() : ''
   360→    const reason =
   361→      typeof r.reason === 'string' && r.reason.trim() ? r.reason.trim() : ''
   362→    if (title && prompt) out.push({ title, prompt, reason })
   363→  })
   364→  if (out.length === 0) return undefined
   365→  return out
   366→}

   368→/** Strip code fences + grab the outermost JSON object. */
   369→function stripFences(s: string): string {
   370→  let out = s.trim()
   371→  out = out.replace(/^```(?:json)?\s*/i, '').replace(/```$/i, '').trim()
   372→  const first = out.indexOf('{')
   373→  const last = out.lastIndexOf('}')
   374→  if (first !== -1 && last !== -1 && last > first) {
   375→    out = out.slice(first, last + 1)
   376→  }
   377→  return out
   378→}

   395→/** Run a web search + LLM synthesis to ground `needs_api` candidates in real
   396→ *  docs. Returns a ResearchResult. Used inline by the chat route when the
   397→ *  intent is needs_api (so the chat response carries both apiDiscovery +
   398→ *  the sources the discovery was based on). */
   399→async function runResearch(query: string): Promise<ResearchResult> {
   400→  let searchResults: SearchResult[] = []
   401→  try {
   402→    const zai = await ZAI.create()
   403→    const raw = await zai.functions.invoke('web_search', {
   404→      query: `${query} API documentation`,
   405→      num: 8,
   406→    })
   407→    if (Array.isArray(raw)) {
   408→      searchResults = raw
   409→        .filter(
   410→          (r) =>
   411→            !!r && typeof r === 'object' && typeof (r as { url?: unknown }).url === 'string',
   412→        )
   413→        .map((r) => {
   414→          const o = r as unknown as Record<string, unknown>
   415→          return {
   416→            url: String(o.url),
   417→            name: typeof o.name === 'string' ? o.name : String(o.url),
   418→            snippet: typeof o.snippet === 'string' ? o.snippet : undefined,
   419→            host_name: typeof o.host_name === 'string' ? o.host_name : undefined,
   420→            rank: typeof o.rank === 'number' ? o.rank : undefined,
   421→          }
   422→        })
   423→    }
   424→  } catch (err) {
   425→    console.error('[api/agent/chat] web_search failed:', err)
   426→  }
   427→
   428→  const sources = searchResults.slice(0, 8).map((r) => ({
   429→    title: r.name || r.url,
   430→    url: r.url,
   431→    snippet: r.snippet || undefined,
   432→  }))
   433→
   434→  let summary =
   435→    sources.length > 0
   436→      ? `I found ${sources.length} source(s) about "${query}".`
   437→      : `I couldn't find web results about "${query}" right now.`
   438→  let candidates: ApiDiscoveryCandidate[] = []
   439→
   440→  const systemPrompt = `You are researching APIs for an automation platform. Given these search results about '${query}', identify any APIs or MCP servers found. For each, return: service name, kind (mcp/api/http), the doc URL, a description, 2-4 plausible tools (id + name + description), and what credentials are needed (as credentialFields). Respond with ONLY JSON: { summary, sources: [{title,url,snippet}], candidates: [ApiDiscoveryCandidate] }. If nothing real was found, return empty candidates with an honest summary. The \`candidates\` array shape is: [{ id, service, kind: 'mcp'|'api'|'http', specUrl?, url?, description, tools: [{id,name,description}], credentialFields: [{key,label,type:'apikey'|'oauth'|'mcp_token',placeholder?,required}] }]. Never invent tools for the user's existing connected tools — only for the discovered service. credentialFields should describe what the user must provide to connect.`
   441→
   442→  const sourcesBlock = sources.length
   443→    ? sources
   444→        .map(
   445→          (s, i) =>
   446→            `${i + 1}. ${s.title}\n   URL: ${s.url}\n   ${s.snippet ? s.snippet : ''}`.trim(),
   447→        )
   448→        .join('\n')
   449→    : '(No web results.)'
   450→
   451→  try {
   452→    const zai = await ZAI.create()
   453→    const completion = await zai.chat.completions.create({
   454→      messages: [
   455→        { role: 'system', content: systemPrompt },
   456→        {
   457→          role: 'user',
   458→          content: `Web search results for "${query} API documentation":\n\n${sourcesBlock}\n\nSynthesize a ResearchResult. Reply with ONLY the JSON object.`,
   459→        },
   460→      ],
   461→      thinking: { type: 'disabled' },
   462→    })
   463→    const text = completion.choices[0]?.message?.content || ''
   464→    const cleaned = stripFences(text)
   465→    const parsed = JSON.parse(cleaned) as {
   466→      summary?: unknown
   467→      candidates?: unknown
   468→    }
   469→    if (typeof parsed.summary === 'string' && parsed.summary.trim()) {
   470→      summary = parsed.summary.trim()
   471→    }
   472→    candidates = coerceApiDiscovery(parsed.candidates) || []
   473→  } catch (err) {
   474→    console.error('[api/agent/chat] research LLM synthesis failed:', err)
   475→  }
   476→
   477→  return { query, summary, sources, candidates }
   478→}

   491→/** Coerce the LLM-proposed WorkflowStep (must have an http field). */
   492→function coerceProposedStep(raw: unknown): WorkflowStep | undefined {
   493→  if (!raw || typeof raw !== 'object') return undefined
   494→  const r = raw as Record<string, unknown>
   495→  const id = typeof r.id === 'string' && r.id ? r.id : 's1'
   496→  const label =
   497→    typeof r.label === 'string' && r.label ? r.label : 'Call API'
   498→  const step: WorkflowStep = { id, kind: 'tool', label }
   499→
   500→  if (typeof r.tool === 'string' && r.tool) step.tool = r.tool
   501→  if (r.inputs && typeof r.inputs === 'object') {
   502→    step.inputs = r.inputs as Record<string, unknown>
   503→  }
   504→
   505→  if (r.http && typeof r.http === 'object') {
   506→    const h = r.http as Record<string, unknown>
   507→    const method =
   508→      typeof h.method === 'string' &&
   509→      ['GET', 'POST', 'PUT', 'PATCH', 'DELETE'].includes(
   510→        String(h.method).toUpperCase(),
   511→      )
   512→        ? (String(h.method).toUpperCase() as
   513→            | 'GET'
   514→            | 'POST'
   515→            | 'PUT'
   516→            | 'PATCH'
   517→            | 'DELETE')
   518→        : 'GET'
   519→    const url = typeof h.url === 'string' ? h.url : ''
   520→    if (!url) return undefined
   521→    const headers =
   522→      h.headers && typeof h.headers === 'object' && !Array.isArray(h.headers)
   523→        ? Object.fromEntries(
   524→            Object.entries(h.headers as Record<string, unknown>)
   525→              .filter(
   526→                ([, v]) => typeof v === 'string' || typeof v === 'number',
   527→              )
   528→              .map(([k, v]) => [k, String(v)]),
   529→          )
   530→        : undefined
   531→    const body = 'body' in h ? h.body : undefined
   532→    const auth =

   541→            } = { type }

   545→            }
   546→            return out

   550→      typeof h.description === 'string' && h.description

   552→        : undefined

   557→      ...(body !== undefined ? { body } : {}),
   558→      ...(auth ? { auth } : {}),
   559→      ...(description ? { description } : {}),
   560→    }
   561→  }
   562→
   563→  if (typeof r.note === 'string' && r.note) step.note = r.note
   564→  return step

   567→/** Analyze an attached script and return a ScriptAnalysis. The reply text
   568→ *  should explain what the script does and propose adding it as a step. */
   569→async function runScriptAnalysis(
   570→  script: string,
   571→  language: 'curl' | 'python' | 'javascript' | 'auto',
   572→): Promise<ScriptAnalysis> {
   573→  const truncated =
   574→    script.length > 4000 ? script.slice(0, 4000) + '\n…(truncated)' : script
   575→
   576→  const systemPrompt = `You are an API reverse-engineer. Given this script, infer what API it calls. Return ONLY JSON: { language, summary, inferredCalls: [{ method, url, headers?, bodyShape?, authType?, description }], proposedStep: WorkflowStep (a tool step with an http field), proposedIntegration?: { name, kind, description, tools } }. The proposedStep should be a ready-to-use workflow step with a unique id, kind 'tool', a clear label, and an http field built from the inferred call. If the script doesn't call an API, say so in the summary and return empty inferredCalls.
   577→
   578→The proposedStep.http shape is:
   579→{ method: 'GET'|'POST'|'PUT'|'PATCH'|'DELETE', url: string, headers?: Record<string,string>, body?: any, auth?: { type: 'bearer'|'apikey_header'|'basic'|'none', ref?: string, headerName?: string }, description?: string }
   580→
   581→For auth:
   582→- bearer: token in Authorization header (set auth.type='bearer', auth.ref='cred_<service>')
   583→- apikey_header: API key in a custom header (set auth.type='apikey_header', auth.headerName='X-Api-Key', auth.ref='cred_<service>')
   584→- basic: HTTP basic auth (auth.type='basic')
   585→- none: no auth (auth.type='none')
   586→
   587→For the proposedIntegration, return { name, kind: 'mcp'|'api'|'http', description, tools: [{id,name,description}] } where the tools match what the script demonstrates. Use a descriptive name based on the URL's host or service.`
   588→
   589→  try {
   590→    const zai = await ZAI.create()
   591→    const completion = await zai.chat.completions.create({
   592→      messages: [
   593→        { role: 'system', content: systemPrompt },
   594→        {
   595→          role: 'user',
   596→          content: `Detected/inferred language hint: ${language}\n\nScript:\n\`\`\`\n${truncated}\n\`\`\`\n\nAnalyze this script and return ONLY the JSON object.`,
   597→        },
   598→      ],
   599→      thinking: { type: 'disabled' },
   600→    })
   601→    const text = completion.choices[0]?.message?.content || ''
   602→    const cleaned = stripFences(text)
   603→    const parsed = JSON.parse(cleaned) as {
   604→      language?: unknown
   605→      summary?: unknown
   606→      inferredCalls?: unknown
   607→      proposedStep?: unknown
   608→      proposedIntegration?: unknown
   609→    }
   610→
   611→    const lang =
   612→      typeof parsed.language === 'string' && parsed.language
   613→        ? parsed.language
   614→        : language === 'auto'
   615→          ? 'unknown'
   616→          : language
   617→    const summary =
   618→      typeof parsed.summary === 'string' && parsed.summary
   619→        ? parsed.summary
   620→        : 'Could not infer what this script does.'
   621→
   622→    const inferredCalls = Array.isArray(parsed.inferredCalls)
   623→      ? parsed.inferredCalls
   624→          .filter(
   625→            (c): c is Record<string, unknown> => !!c && typeof c === 'object',
   626→          )
   627→          .map((c) => {
   628→            const method =
   629→              typeof c.method === 'string' ? c.method.toUpperCase() : 'GET'
   630→            const url = typeof c.url === 'string' ? c.url : ''
   631→            const headers =
   632→              c.headers && typeof c.headers === 'object' && !Array.isArray(c.headers)
   633→                ? Object.fromEntries(
   634→                    Object.entries(c.headers as Record<string, unknown>)
   635→                      .filter(
   636→                        ([, v]) => typeof v === 'string' || typeof v === 'number',
   637→                      )
   638→                      .map(([k, v]) => [k, String(v)]),
   639→                  )
   640→                : undefined
   641→            const bodyShape =
   642→              typeof c.bodyShape === 'string' ? c.bodyShape : undefined
   643→            const authType = coerceAuthType(c.authType)
   644→            const description =
   645→              typeof c.description === 'string' ? c.description : ''
   646→            return {
   647→              method,
   648→              url,
   649→              ...(headers ? { headers } : {}),
   650→              ...(bodyShape ? { bodyShape } : {}),
   651→              authType,
   652→              description,
   653→            }
   654→          })
   655→      : []
   656→
   657→    const proposedStep = coerceProposedStep(parsed.proposedStep)
   658→    const proposedIntegration =
   659→      parsed.proposedIntegration &&
   660→      typeof parsed.proposedIntegration === 'object'
   661→        ? (() => {
   662→            const pi = parsed.proposedIntegration as Record<string, unknown>
   663→            const name =
   664→              typeof pi.name === 'string' && pi.name ? pi.name : ''
   665→            if (!name) return undefined
   666→            const kind = coerceKind(pi.kind)
   667→            const description =
   668→              typeof pi.description === 'string' ? pi.description : ''
   669→            const tools = Array.isArray(pi.tools)
   670→              ? pi.tools
   671→                  .filter(
   672→                    (t): t is Record<string, unknown> =>
   673→                      !!t && typeof t === 'object',
   674→                  )
   675→                  .slice(0, 6)
   676→                  .map((t, j) => ({
   677→                    id:
   678→                      typeof t.id === 'string' && t.id
   679→                        ? t.id
   680→                        : `tool${j + 1}`,
   681→                    name:
   682→                      typeof t.name === 'string' && t.name
   683→                        ? t.name
   684→                        : `Tool ${j + 1}`,
   685→                    description:
   686→                      typeof t.description === 'string' ? t.description : '',
   687→                  }))
   688→              : []
   689→            return { name, kind, description, tools }
   690→          })()
   691→        : undefined
   692→
   693→    return {
   694→      language: lang,
   695→      summary,
   696→      inferredCalls,
   697→      ...(proposedStep ? { proposedStep } : {}),
   698→      ...(proposedIntegration ? { proposedIntegration } : {}),
   699→    }
   700→  } catch (err) {
   701→    console.error('[api/agent/chat] script analysis failed:', err)
   702→    return {
   703→      language: language === 'auto' ? 'unknown' : language,
   704→      summary:
   705→        "I tried to analyze the attached script but couldn't reach the reasoning engine. Please try again in a moment.",
   706→      inferredCalls: [],
   707→    }
   708→  }
   709→}

   806→    if (a.lastRun) {
   807→      const minsAgo = Math.max(
   808→        1,
   809→        Math.round((Date.now() - a.lastRun.startedAt.getTime()) / 60000),
   810→      )
   811→      const ago = minsAgo < 60 ? `${minsAgo}m ago` : `${Math.round(minsAgo / 60)}h ago`
   812→      lines.push(
   813→        `   Last run (${ago}, status: ${a.lastRun.status}): ${a.lastRun.reportSummary || `Processed ${a.lastRun.itemsProcessed} items.`} — ${a.lastRun.itemsProcessed} items, ${a.lastRun.automaticCount} automatic, ${a.lastRun.flaggedCount} flagged.`,
   814→      )
   815→    } else {
   816→      lines.push(`   Last run: (none yet)`)
   817→    }
   818→    lines.push(`   Current steps:`)
   819→    lines.push(renderStepsBrief(steps).replace(/^/gm, '   '))
   820→    return lines.join('\n')
   821→  })
   822→  return blocks.join('\n\n')
   823→}

  1359→    // For empty messages, prompt the LLM to return suggestions.
  1360→    const userMessage = isEmpty
  1361→      ? `The user just opened the chat. Look at their profile, existing agents, and data sources, and suggest 3-4 tailored automation ideas. Return intent "general" with a short \`reply\` (one sentence, friendly) and a \`suggestions\` array of 3-4 items, each with title + prompt + reason. Pick ideas that FILL GAPS — things they don't already have an agent for but would benefit from given their data sources.`
  1362→      : ((): string => {
  1363→          // Condense prior history for the model.
  1364→          const priorTurns = (body.history || [])
  1365→            .filter((m) => m.role === 'user' && m.content)
  1366→            .slice(-4)
  1367→            .map((m) => `- ${m.content}`)
  1368→            .join('\n')
  1369→          return priorTurns
  1370→            ? `Earlier in this conversation the user discussed:\n${priorTurns}\n\nNow the user says:\n${message}`
  1371→            : message
  1372→        })()
  1373→
  1374→    let parsed: {
  1375→      trace?: { label: string; detail?: string }[]
  1376→      reply?: string
  1377→      intent?: string
  1378→      title?: string
  1379→      workflowProposal?: {
  1380→        name?: string
  1381→        description?: string
  1382→        department?: string
  1383→        title?: string
  1384→        steps?: unknown
  1385→      }
  1386→      switchToAgentId?: string
  1387→      editingAgentId?: string
  1388→      clarification?: unknown
  1389→      apiDiscovery?: unknown
  1390→      suggestions?: unknown
  1391→    } | null = null
  1392→
  1393→    try {
  1394→      const zai = await ZAI.create()
  1395→      const completion = await zai.chat.completions.create({
  1396→        messages: [
  1397→          { role: 'system', content: systemPrompt },
  1398→          { role: 'user', content: userMessage },
  1399→        ],
  1400→        thinking: model === 'thinking' ? { type: 'enabled' } : { type: 'disabled' },
  1401→        stream: false,
  1402→      })
  1403→      const text = completion.choices[0]?.message?.content || ''
  1404→      const cleaned = stripFences(text)
  1405→      parsed = JSON.parse(cleaned) as typeof parsed
  1406→    } catch (err) {
  1407→      console.error('[api/agent/chat] LLM call/parse failed:', err)
  1408→      // Fall through to fallback below.
  1409→    }
  1410→
  1411→    // Empty-message path: always return suggestions (fallback if LLM failed).
  1412→    if (isEmpty) {
  1413→      const suggestions = parsed ? coerceSuggestions(parsed.suggestions) : undefined
  1414→      return NextResponse.json({
  1415→        reply:
  1416→          parsed?.reply ||
  1417→          "Hi — I'm your Apical Assistant. Tell me what you'd like to automate, or pick one of these to get started:",
  1418→        trace: Array.isArray(parsed?.trace)
  1419→          ? parsed!.trace
  1420→              .filter((t) => t && typeof t.label === 'string')
  1421→              .slice(0, 5)
  1422→              .map((t) => ({ label: t.label, detail: t.detail }))
  1423→          : [{ label: 'Looked around', detail: 'Read your profile and existing agents.' }],
  1424→        intent: 'general',
  1425→        suggestions:
  1426→          suggestions ||
  1427→          (profile
  1428→            ? [
  1429→                {
  1430→                  title: 'Sort my scanner PDFs',
  1431→                  prompt:
  1432→                    "Sort the PDFs my scanner dumps into /Scan Inbox by client, and file them. Ask me if anything is unclear.",
  1433→                  reason: 'You scan paper docs daily — a filing agent saves an hour a week.',
  1434→                },
  1435→                {
  1436→                  title: 'Chase overdue invoices',
  1437→                  prompt:
  1438→                    'Check unpaid invoices every day. Send a polite reminder if 7 days late; if 30 days, draft an escalation for me to approve.',
  1439→                  reason: `You use ${profile.dataSources.some((d) => d.label.toLowerCase().includes('stripe')) ? 'Stripe' : 'your billing system'} — automate the chase for cash flow.`,
  1440→                },
  1441→                {
  1442→                  title: 'Weekly client updates',
  1443→                  prompt:
  1444→                    'Every Monday, draft a short summary email to each client about last week. Send me the drafts first.',
  1445→                  reason: 'Recurring client comms are a perfect first agent.',
  1446→                },
  1447→              ]
  1448→            : undefined),
  1449→      } satisfies AgentResponse)
  1450→    }
  1451→
  1452→    // Non-empty message but no message validation:
  1453→    if (!isEmpty && !message) {
  1454→      return NextResponse.json(
  1455→        { error: 'message is required' },
  1456→        { status: 400 },
  1457→      )
  1458→    }
  1459→
  1460→    if (!parsed || typeof parsed.reply !== 'string') {
  1461→      return NextResponse.json(
  1462→        fallbackResponse(message, { mentionIds }),
  1463→      )
  1464→    }
  1465→
  1466→    // Coerce intent to one of the five valid values.
  1467→    const validIntents: Intent[] = [
  1468→      'new_agent',
  1469→      'edit_existing',
  1470→      'general',
  1471→      'needs_clarification',
  1472→      'needs_api',
  1473→    ]
  1474→    const intent: Intent = validIntents.includes(parsed.intent as Intent)
  1475→      ? (parsed.intent as Intent)
  1476→      : 'general'
  1477→
  1478→    const trace = Array.isArray(parsed.trace)
  1479→      ? parsed.trace
  1480→          .filter((t) => t && typeof t.label === 'string')
  1481→          .slice(0, 5)
  1482→          .map((t) => ({ label: t.label, detail: t.detail }))
  1483→      : []
  1484→
  1485→    const response: AgentResponse = {
  1486→      reply: parsed.reply,
  1487→      trace,
  1488→      intent,
  1489→    }
  1490→
  1491→    if (typeof parsed.title === 'string' && parsed.title.trim()) {
  1492→      response.title = parsed.title.trim()
  1493→    }
  1494→
  1495→    if (intent === 'new_agent') {
  1496→      const proposal = parsed.workflowProposal
  1497→      const steps = normalizeProposalSteps(proposal?.steps)
  1498→      if (steps.length === 0) {
  1499→        return NextResponse.json(
  1500→          fallbackResponse(message, { mentionIds }),
  1501→        )
  1502→      }
  1503→      // Pick the agent name from the user's preferred style (UserProfile.agentNameStyle).
  1504→      // The LLM still picks the department, title, and steps — but the NAME comes
  1505→      // from generateAgentName, not the LLM, so names are always in the user's style.
  1506→      const style: AgentNameStyle =
  1507→        profileRow?.agentNameStyle === 'evocative' ? 'evocative' : 'descriptive'
  1508→      const existingNames = agents.map((a) => a.name)
  1509→      const name = await pickAgentName(
  1510→        style,
  1511→        proposal?.name,
  1512→        message || proposal?.description || '',
  1513→        existingNames,
  1514→      )
  1515→      response.workflowProposal = {
  1516→        name,
  1517→        description:
  1518→          typeof proposal?.description === 'string' && proposal.description.trim()
  1519→            ? proposal.description.trim()
  1520→            : 'A new agent for your workspace.',
  1521→        department:
  1522→          typeof proposal?.department === 'string' && proposal.department.trim()
  1523→            ? proposal.department.trim()
  1524→            : 'General',
  1525→        title:
  1526→          typeof proposal?.title === 'string' && proposal.title.trim()
  1527→            ? proposal.title.trim()
  1528→            : 'Agent',
  1529→        steps: { version: 1, steps },
  1530→      }
  1531→    } else if (intent === 'edit_existing') {
  1532→      const validIds = new Set(agents.map((a) => a.id))
  1533→      const mentionedSet = new Set(mentionIds)
  1534→      const llmSwitching =
  1535→        typeof parsed.switchToAgentId === 'string' &&
  1536→        validIds.has(parsed.switchToAgentId)
  1537→          ? parsed.switchToAgentId
  1538→          : undefined
  1539→      const llmEditing =
  1540→        typeof parsed.editingAgentId === 'string' &&
  1541→        validIds.has(parsed.editingAgentId)
  1542→          ? parsed.editingAgentId
  1543→          : undefined
  1544→
  1545→      // @mention model:
  1546→      // - If the LLM picked a mentioned agent as editing → use it.
  1547→      // - If the LLM picked a non-mentioned agent as editing → that's a switch
  1548→      //   (the user tagged A but the LLM thinks they mean B).
  1549→      // - If the LLM picked a switch → use it (explicit switch).
  1550→      // - If neither was set but there's exactly one mention → edit that one.
  1551→      // - If neither was set but there are multiple mentions → leave it to the
  1552→      //   frontend to disambiguate (no editing/switch id).
  1553→      let switching = llmSwitching
  1554→      let editing: string | undefined
  1555→      if (llmEditing) {
  1556→        if (mentionedSet.has(llmEditing)) {
  1557→          editing = llmEditing
  1558→        } else if (!switching) {
  1559→          switching = llmEditing
  1560→        } else {
  1561→          editing = llmEditing
  1562→        }
  1563→      }
  1564→      if (!switching && !editing) {
  1565→        if (mentionedSet.size === 1) {
  1566→          editing = mentionIds[0]
  1567→        } else if (mentionedSet.size > 1) {
  1568→          // Multiple mentions, no explicit pick — demote to general so the
  1569→          // UI asks the user to clarify which agent they mean.
  1570→          response.intent = 'general'
  1571→        }
  1572→      }
  1573→
  1574→      if (response.intent === 'edit_existing') {
  1575→        if (switching) response.switchToAgentId = switching
  1576→        else if (editing) response.editingAgentId = editing
  1577→        else {
  1578→          // Couldn't resolve an agent — demote to general so the UI doesn't dead-end.
  1579→          response.intent = 'general'
  1580→        }
  1581→      }
  1582→    } else if (intent === 'needs_clarification') {
  1583→      const clar = coerceClarification(parsed.clarification)
  1584→      if (!clar) {
  1585→        // No valid clarification — demote to general so the UI doesn't dead-end.
  1586→        response.intent = 'general'
  1587→      } else {
  1588→        response.clarification = clar
  1589→      }
  1590→    } else if (intent === 'needs_api') {
  1591→      let candidates = coerceApiDiscovery(parsed.apiDiscovery)
  1592→      // Ground the discovery in real web research. Use the user's message
  1593→      // (or the first candidate's service name) as the query.
  1594→      const researchQuery =
  1595→        message ||
  1596→        (candidates && candidates[0]?.service) ||
  1597→        'API'
  1598→      let research: ResearchResult | undefined
  1599→      try {
  1600→        research = await runResearch(researchQuery)
  1601→      } catch (err) {
  1602→        console.error('[api/agent/chat] runResearch failed:', err)
  1603→      }
  1604→
  1605→      // Prefer the LLM's own apiDiscovery (it's grounded in the user's intent
  1606→      // + the workspace context). If the LLM produced no candidates but the
  1607→      // research surfaced some, use those instead.
  1608→      if ((!candidates || candidates.length === 0) && research?.candidates?.length) {
  1609→        candidates = research.candidates
  1610→      }
  1611→      if (!candidates || candidates.length === 0) {
  1612→        response.intent = 'general'
  1613→      } else {
  1614→        response.apiDiscovery = candidates
  1615→        if (research) response.research = research
  1616→      }
  1617→    } else if (intent === 'needs_research') {
  1618→      // Deep research: crawl the web, find data sources, propose a workflow.
  1619→      try {
  1620→        const researchResp = await fetch(`${new URL(req.url).origin}/api/agent/research`, {
  1621→          method: 'POST',
  1622→          headers: { 'Content-Type': 'application/json' },
  1623→          body: JSON.stringify({ goal: message, context: undefined }),
  1624→        })
  1625→        if (researchResp.ok) {
  1626→          const plan = await researchResp.json()
  1627→          response.researchPlan = plan
  1628→          response.reply = response.reply || `I researched "${message}" and found ${plan.findings?.length ?? 0} potential data sources. Here's my proposed strategy and workflow — review it below.`
  1629→        }
  1630→      } catch (err) {
  1631→        console.error('[api/agent/chat] deep research failed:', err)
  1632→        response.reply = `I tried to research "${message}" but hit a snag. Could you try rephrasing, or give me more detail about what you're looking for?`
  1633→        response.intent = 'general'
  1634→      }
  1635→    } else if (intent === 'general') {
  1636→      // Optional suggestions on a general turn.
  1637→      const suggestions = coerceSuggestions(parsed.suggestions)
  1638→      if (suggestions) response.suggestions = suggestions
  1639→    }
  1640→
  1641→    return NextResponse.json(response)
  1642→  } catch (err) {
  1643→    console.error('[api/agent/chat] failed:', err)
  1644→    return NextResponse.json(
  1645→      fallbackResponse('', { mentionIds: [] }),
  1646→    )
  1647→  }
  1648→}
