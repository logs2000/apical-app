    46→// ─── OS detection ──────────────────────────────────────────────────────────
    47→
    48→type DetectedOS = "mac" | "windows" | "linux" | "other";
    49→
    50→function detectOS(): DetectedOS {
    51→  if (typeof navigator === "undefined") return "other";
    52→  const ua = navigator.userAgent.toLowerCase();
    53→  if (ua.includes("mac os x") || ua.includes("darwin")) return "mac";
    54→  if (ua.includes("windows")) return "windows";
    55→  if (ua.includes("linux")) return "linux";
    56→  return "other";
    57→}
    58→function osLabel(os: DetectedOS): string {
    59→  return os === "mac" ? "macOS" : os === "windows" ? "Windows" : os === "linux" ? "Linux" : "Pick platform";
    60→}
    61→function downloadButtonLabel(os: DetectedOS): string {
    62→  return os === "other" ? "Download" : `Download for ${osLabel(os)}`;
    63→}
    64→function downloadUrl(os: DetectedOS): string {
    65→  const base = "/downloads";
    66→  if (os === "mac") return `${base}/apical-mac.dmg`;
    67→  if (os === "windows") return `${base}/apical-windows.exe`;
    68→  if (os === "linux") return `${base}/apical-linux.AppImage`;
    69→  return `${base}/`;
    70→}

    72→  if (os === "mac") return "brew install --cask apical";

    74→  if (os === "linux") return "curl -fsSL https://apical.dev/install.sh | sh";
    75→  return "curl -fsSL https://apical.dev/install.sh | sh";
    76→}

    86→// ─── Reduced motion hook (replaces framer-motion useReducedMotion) ────────
    87→
    88→function usePrefersReducedMotion(): boolean {
    89→  const [reduced, setReduced] = React.useState(false);
    90→
    91→  React.useEffect(() => {
    92→    const mql = window.matchMedia("(prefers-reduced-motion: reduce)");
    93→    setReduced(mql.matches);
    94→    const handler = (e: MediaQueryListEvent) => setReduced(e.matches);
    95→    mql.addEventListener("change", handler);
    96→    return () => mql.removeEventListener("change", handler);
    97→  }, []);
    98→
    99→  return reduced;
   100→}

   102→// ─── Intersection observer hook for scroll-triggered animations ───────────
   103→
   104→function useInView(options?: IntersectionObserverInit) {
   105→  const ref = React.useRef<HTMLDivElement>(null);
   106→  const [inView, setInView] = React.useState(false);
   107→
   108→  React.useEffect(() => {
   109→    const el = ref.current;
   110→    if (!el) return;
   111→
   112→    const observer = new IntersectionObserver(([entry]) => {
   113→      if (entry.isIntersecting) {
   114→        setInView(true);
   115→        observer.unobserve(el);
   116→      }
   117→    }, options);
   118→
   119→    observer.observe(el);
   120→    return () => observer.disconnect();
   121→  }, [options]);
   122→
   123→  return { ref, inView };
   124→}

   126→// ─── AnimatedCard — wrapper component that handles scroll-triggered animation ─

   128→function AnimatedCard({
   129→  children,
   130→  reduced,
   131→  delayMs = 0,
   132→  className,
   133→  rootMargin = "-60px",
   134→  translateY = 12,
   135→}: {
   136→  children: React.ReactNode;
   137→  reduced: boolean;
   138→  delayMs?: number;
   139→  className?: string;
   140→  rootMargin?: string;
   141→  translateY?: number;
   142→}) {
   143→  const { ref, inView } = useInView({ threshold: 0.05, rootMargin });
   144→  return (
   145→    <div
   146→      ref={ref}
   147→      className={cn("transition-all duration-500", className)}
   148→      style={{
   149→        opacity: reduced ? 1 : inView ? 1 : 0,
   150→        transform: reduced ? "none" : inView ? "none" : `translateY(${translateY}px)`,
   151→        transitionDelay: reduced ? undefined : `${delayMs}ms`,
   152→      }}
   153→    >
   154→      {children}
   155→    </div>
   156→  );
   157→}

   869→
   870→function computePrice(
   871→  plan: Plan,
   872→  interval: "monthly" | "yearly",
   873→): { display: string; suffix?: string; sub?: string } {
   874→  if (plan.id === "enterprise") return { display: "Custom", sub: "Volume pricing — talk to sales." };
   875→  if (plan.id === "free") return { display: "$0", suffix: "/mo", sub: "Free forever. No credit card." };
   876→  const perSeat = plan.id === "team" ? " /seat" : "";
   877→  if (interval === "monthly") {
   878→    return {
   879→      display: `$${plan.priceMonthly}`,
   880→      suffix: `/mo${perSeat}`,
   881→      sub: plan.id === "team" ? "5 seats included." : "Billed monthly.",
   882→    };
   883→  }
   884→  const perMonth = plan.priceYearly / 12;
   885→  const monthsFree = 12 - Math.round(plan.priceYearly / plan.priceMonthly);
   886→  return {
   887→    display: `$${perMonth % 1 === 0 ? perMonth : perMonth.toFixed(2)}`,
   888→    suffix: `/mo${perSeat}`,
   889→    sub: `$${plan.priceYearly}/yr${perSeat} — ${monthsFree} months free`,
   890→  };
   891→}

  1000→function DownloadButton({
  1001→  os,
  1002→  variant = "default",
  1003→  size = "default",
  1004→  className,
  1005→}: {
  1006→  os: DetectedOS;
  1007→  variant?: "default" | "outline" | "ghost";
  1008→  size?: "default" | "sm" | "lg";
  1009→  className?: string;
  1010→}) {
  1011→  const [open, setOpen] = React.useState(false);
  1012→  const { toast } = useToast();
  1013→
  1014→  const handleClick = async () => {
  1015→    if (os === "other") {
  1016→      setOpen(true);
  1017→      return;
  1018→    }
  1019→    try {
  1020→      const res = await fetch(downloadUrl(os), { method: "HEAD" });
  1021→      if (res.ok) {
  1022→        window.location.href = downloadUrl(os);
  1023→        return;
  1024→      }
  1025→      setOpen(true);
  1026→    } catch {
  1027→      setOpen(true);
  1028→    }
  1029→  };
  1030→
  1031→  return (
  1032→    <>
  1033→      <Button variant={variant} size={size} onClick={handleClick} className={className}>
  1034→        <Download className="mr-1.5 h-4 w-4" />
  1035→        {downloadButtonLabel(os)}
  1036→      </Button>
  1037→      <DownloadDialog open={open} onOpenChange={setOpen} os={os} onCopied={() => toast({ title: "Copied" })} />
  1038→    </>
  1039→  );
  1040→}

  1042→function DownloadDialog({
  1043→  open,
  1044→  onOpenChange,
